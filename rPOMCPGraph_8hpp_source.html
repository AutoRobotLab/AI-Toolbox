117  void BeliefNode<true>::updateBeliefAndKnowledge(size_t s) {
118  // Remove entropy term for this state from summatory
119  knowledgeMeasure_ -= trackBelief_[s].negativeEntropy;
120  // Updating belief
121  trackBelief_[s].N += 1;
122  // Computing new entropy term for this state
123  double p = static_cast<double>(trackBelief_[s].N) / static_cast<double>(N+1);
124  double newEntropy = p * std::log(p);
125  // Update values
126  trackBelief_[s].negativeEntropy = newEntropy;
127  knowledgeMeasure_ += newEntropy;
