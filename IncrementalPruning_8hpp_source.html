93  template <typename M, typename std::enable_if<is_model<M>::value, int>::type>
94  boost::multi_array<VList, 2> makeAllProjs(const M & model, const VList & w) {
95  size_t S = model.getS(), A = model.getA(), O = model.getO();
96  double discount = model.getDiscount();
97 
98  boost::multi_array<VList, 2> projections( boost::extents[A][O] );
99 
100  static auto immediateRewards = computeImmediateRewards(model);
101  static auto possibleObservations = computePossibleObservations(model);
102 
103  for ( size_t a = 0; a < A; ++a ) {
104  for ( size_t o = 0; o < O; ++o ) {
105  // Here we put in just the immediate rewards so that the cross-summing step in the main
106  // function works correctly. However we communicate via the boolean that pruning should
107  // not be done at this step (since adding constants shouldn't do anything anyway).
108  if ( !possibleObservations[a][o] ) { projections[a][o].push_back(immediateRewards); continue; }
109 
110  for ( auto & v : w ) {
111  // For each value function in the previous timestep, we compute the new value
112  // if we performed action a and obtained observation o.
113  MDP::ValueFunction vproj(S, 0.0);
114  for ( size_t s = 0; s < S; ++s ) {
115  // vproj_{a,o}[s] = R(s,a) / |O| + discount * sum_{s'} ( T(s,a,s') * O(s',a,o) * v_{t-1}(s') )
116  for ( size_t s1 = 0; s1 < S; ++s1 )
117  vproj[s] += model.getTransitionProbability(s,a,s1) * model.getObservationProbability(s1,a,o) * v[s1];
118 
119  vproj[s] *= discount;
120  vproj[s] += immediateRewards[s] / O;
121  }
122  projections[a][o].push_back(v);
123  }
124  }
125  }
126  return projections;
127  }

130  Table2D computeImmediateRewards(const M & model) {
131  size_t S = model.getS(), A = model.getA(), O = model.getO();
132  Table2D immediateRewards( boost::extents[A][S] );
133 
134  for ( size_t a = 0; a < A; ++a ) {
135  for ( size_t s = 0; s < S; ++s ) {
136  for ( size_t s1 = 0; s1 < S; ++s1 )
137  immediateRewards[a][s] += model.getTransitionProbability(s,a,s1) * model.getExpectedReward(s,a,s1);
138 
139  // You can find out why this is divided in the incremental pruning paper =)
140  // The idea is that at the end of all the cross sums it's going to add up to the correct value.
141  immediateRewards[a][s] /= static_cast<double>(O);
142  }
143  }
