46  std::tuple<bool, ValueFunction> IncrementalPruning::operator()(const M & model) {
47  // Initialize "global" variables
48  S = model.getS();
49  A = model.getA();
50  O = model.getO();
51 
52  auto possibleObservations = computePossibleObservations(model);
53  auto immediateRewards = computeImmediateRewards(model);
54 
55  ValueFunction v(1, VList(1, makeVEntry(S))); // TODO: May take user input
56 
57  unsigned timestep = 1;
58 
59  Pruner prune(S);
60 
61  // And off we go
62  while ( timestep <= horizon_ ) {
63  // Compute all possible outcomes, from our previous results.
64  // This means that for each action-observation pair, we are going
65  // to obtain the same number of possible outcomes as the number
66  // of entries in our initial vector w.
67  auto projs = makeAllProjections(model, v[timestep-1], possibleObservations, immediateRewards);
68 
69  size_t finalWSize = 0;
70  // In this method we split the work by action, which will then
71  // be joined again at the end of the loop.
72  for ( size_t a = 0; a < model.getA(); ++a ) {
73  // We prune each outcome separately to be sure
74  // we do not replicate work later.
75  for ( size_t o = 0; o < model.getO(); ++o ) {
76  prune( &projs[a][o] );
77  }
78 
79  for ( size_t o = 1; o < model.getO(); ++o ) {
80  projs[a][0] = crossSum( projs[a][0], projs[a][o], a, o );
81  prune( &projs[a][0] );
82  }
83  finalWSize += projs[a][0].size();
84  }
85  VList w;
86  w.reserve(finalWSize);
87 
88  for ( size_t a = 0; a < model.getA(); ++a )
89  std::move(std::begin(projs[a][0]), std::end(projs[a][0]), std::back_inserter(w));
90 
91  // We have them all, and we prune one final time to be sure we have
92  // computed the parsimonious set of value functions.
93  prune( &w );
94 
95  v.emplace_back(std::move(w));
96 
97  ++timestep;
98  }
99 
100  return std::make_tuple(true, v);
101  }

104  IncrementalPruning::ProjectionsTable IncrementalPruning::makeAllProjections(const M & model, const VList & w,
105  const PossibleObservationsTable & possibleObservations,
106  const Table2D & immediateRewards)
107  {
108  size_t S = model.getS(), A = model.getA(), O = model.getO();
109  double discount = model.getDiscount();
110 
111  ProjectionsTable projections( boost::extents[A][O] );
112 
113  for ( size_t a = 0; a < A; ++a ) {
114  for ( size_t o = 0; o < O; ++o ) {
115  // Here we put in just the immediate rewards so that the cross-summing step in the main
116  // function works correctly. However we communicate via the boolean that pruning should
117  // not be done at this step (since adding constants shouldn't do anything anyway).
118  if ( !possibleObservations[a][o] ) {
119  MDP::Values vproj(S, 0.0);
120  for ( size_t s = 0; s < S; ++s )
121  vproj[s] += immediateRewards[a][s];
122  // We add a parent id anyway in order to keep the code that cross-sums simple. However
123  // note that this fake ID of 0 should never be used, so it should be safe to avoid
124  // setting it to a special value like -1. If one really wants to check, he/she can
125  // just look at the observation table and the belief and see if it makes sense.
126  projections[a][o].emplace_back(std::move(vproj), a, VObs(1,0));
127  continue;
128  }
129 
130  // Otherwise we compute a projection for each ValueFunction supplied to us.
131  for ( size_t i = 0; i < w.size(); ++i ) {
132  auto & v = std::get<VALUES>(w[i]);
133  MDP::Values vproj(S, 0.0);
134  // For each value function in the previous timestep, we compute the new value
135  // if we performed action a and obtained observation o.
136  for ( size_t s = 0; s < S; ++s ) {
137  // vproj_{a,o}[s] = R(s,a) / |O| + discount * sum_{s'} ( T(s,a,s') * O(s',a,o) * v_{t-1}(s') )
138  for ( size_t s1 = 0; s1 < S; ++s1 )
139  vproj[s] += model.getTransitionProbability(s,a,s1) * model.getObservationProbability(s1,a,o) * v[s1];
140 
141  vproj[s] *= discount;
142  vproj[s] += immediateRewards[a][s];
143  }
144  // Set new projection with found value and previous V id.
145  projections[a][o].emplace_back(std::move(vproj), a, VObs(1,i));
146  }
147  }
148  }
149  return projections;
150  }

153  Table2D IncrementalPruning::computeImmediateRewards(const M & model) {
154  Table2D immediateRewards( boost::extents[A][S] );
155 
156  for ( size_t a = 0; a < A; ++a ) {
157  for ( size_t s = 0; s < S; ++s ) {
158  for ( size_t s1 = 0; s1 < S; ++s1 )
159  immediateRewards[a][s] += model.getTransitionProbability(s,a,s1) * model.getExpectedReward(s,a,s1);
160 
161  // You can find out why this is divided in the incremental pruning paper =)
162  // The idea is that at the end of all the cross sums it's going to add up to the correct value.
163  immediateRewards[a][s] /= static_cast<double>(O);
164  }
165  }
166  return immediateRewards;
