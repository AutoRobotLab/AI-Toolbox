134  std::tuple<bool, ValueFunction> IncrementalPruning::operator()(const M & model) {
135  // Initialize "global" variables
136  S = model.getS();
137  A = model.getA();
138  O = model.getO();
139 
140  auto possibleObservations = computePossibleObservations(model);
141  auto immediateRewards = computeImmediateRewards(model);
142 
143  ValueFunction v(1, VList(1, makeVEntry(S))); // TODO: May take user input
144 
145  unsigned timestep = 1;
146 
147  Pruner prune(S);
148 
149  // And off we go
150  while ( timestep <= horizon_ ) {
151  // Compute all possible outcomes, from our previous results.
152  // This means that for each action-observation pair, we are going
153  // to obtain the same number of possible outcomes as the number
154  // of entries in our initial vector w.
155  auto projs = makeAllProjections(model, v[timestep-1], possibleObservations, immediateRewards);
156 
157  size_t finalWSize = 0;
158  // In this method we split the work by action, which will then
159  // be joined again at the end of the loop.
160  for ( size_t a = 0; a < model.getA(); ++a ) {
161  // We prune each outcome separately to be sure
162  // we do not replicate work later.
163  for ( size_t o = 0; o < model.getO(); ++o ) {
164  prune( &projs[a][o] );
165  }
166 
167  for ( size_t o = 1; o < model.getO(); ++o ) {
168  projs[a][0] = crossSum( projs[a][0], projs[a][o], a, o );
169  prune( &projs[a][0] );
170  }
171  finalWSize += projs[a][0].size();
172  }
173  VList w;
174  w.reserve(finalWSize);
175 
176  for ( size_t a = 0; a < model.getA(); ++a )
177  std::move(std::begin(projs[a][0]), std::end(projs[a][0]), std::back_inserter(w));
178 
179  // We have them all, and we prune one final time to be sure we have
180  // computed the parsimonious set of value functions.
181  prune( &w );
182 
183  v.emplace_back(std::move(w));
184 
185  ++timestep;
186  }
187 
188  return std::make_tuple(true, v);
189  }

191  template <typename M, typename std::enable_if<is_model<M>::value, int>::type>
192  IncrementalPruning::ProjectionsTable IncrementalPruning::makeAllProjections(const M & model, const VList & w,
193  const PossibleObservationsTable & possibleObservations,
194  const Table2D & immediateRewards)
195  {
196  size_t S = model.getS(), A = model.getA(), O = model.getO();
197  double discount = model.getDiscount();
198 
199  ProjectionsTable projections( boost::extents[A][O] );
200 
201  for ( size_t a = 0; a < A; ++a ) {
202  for ( size_t o = 0; o < O; ++o ) {
203  // Here we put in just the immediate rewards so that the cross-summing step in the main
204  // function works correctly. However we communicate via the boolean that pruning should
205  // not be done at this step (since adding constants shouldn't do anything anyway).
206  if ( !possibleObservations[a][o] ) {
207  MDP::Values vproj(S, 0.0);
208  for ( size_t s = 0; s < S; ++s )
209  vproj[s] += immediateRewards[a][s];
210  // We add a parent id anyway in order to keep the code that cross-sums simple. However
211  // note that this fake ID of 0 should never be used, so it should be safe to avoid
212  // setting it to a special value like -1. If one really wants to check, he/she can
213  // just look at the observation table and the belief and see if it makes sense.
214  projections[a][o].emplace_back(std::move(vproj), a, VObs(1,0));
215  continue;
216  }
217 
218  // Otherwise we compute a projection for each ValueFunction supplied to us.
219  for ( size_t i = 0; i < w.size(); ++i ) {
220  auto & v = std::get<VALUES>(w[i]);
221  MDP::Values vproj(S, 0.0);
222  // For each value function in the previous timestep, we compute the new value
223  // if we performed action a and obtained observation o.
224  for ( size_t s = 0; s < S; ++s ) {
225  // vproj_{a,o}[s] = R(s,a) / |O| + discount * sum_{s'} ( T(s,a,s') * O(s',a,o) * v_{t-1}(s') )
226  for ( size_t s1 = 0; s1 < S; ++s1 )
227  vproj[s] += model.getTransitionProbability(s,a,s1) * model.getObservationProbability(s1,a,o) * v[s1];
228 
229  vproj[s] *= discount;
230  vproj[s] += immediateRewards[a][s];
231  }
232  // Set new projection with found value and previous V id.
233  projections[a][o].emplace_back(std::move(vproj), a, VObs(1,i));
234  }
235  }
236  }
237  return projections;
238  }

241  Table2D IncrementalPruning::computeImmediateRewards(const M & model) {
242  Table2D immediateRewards( boost::extents[A][S] );
243 
244  for ( size_t a = 0; a < A; ++a ) {
245  for ( size_t s = 0; s < S; ++s ) {
246  for ( size_t s1 = 0; s1 < S; ++s1 )
247  immediateRewards[a][s] += model.getTransitionProbability(s,a,s1) * model.getExpectedReward(s,a,s1);
248 
249  // You can find out why this is divided in the incremental pruning paper =)
250  // The idea is that at the end of all the cross sums it's going to add up to the correct value.
251  immediateRewards[a][s] /= static_cast<double>(O);
252  }
253  }
254  return immediateRewards;
255  }
