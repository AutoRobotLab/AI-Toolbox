134  // This means that for each action-observation pair, we are going
135  // to obtain the same number of possible outcomes as the number
136  // of entries in our initial vector w.

138

142  for ( size_t a = 0; a < A; ++a ) {
143  // We prune each outcome separately to be sure
144  // we do not replicate work later.
145  for ( size_t o = 0; o < O; ++o )
146  prune( &projs[a][o] );
147 
148  for ( size_t o = 1; o < O; ++o ) {
