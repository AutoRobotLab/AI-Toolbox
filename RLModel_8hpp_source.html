299  void RLModel<E>::sync(const size_t s, const size_t a) {
300  // Nothing to do
301  const auto visitSum = experience_.getVisitsSum(s, a);
302  if ( visitSum == 0ul ) return;
303 
304  // Create reciprocal for fast division
305  const double visitSumReciprocal = 1.0 / visitSum;
306 
307  // Normalize
308  for ( size_t s1 = 0; s1 < S; ++s1 ) {
309  const auto visits = experience_.getVisits(s, a, s1);
310  transitions_[a](s, s1) = static_cast<double>(visits) * visitSumReciprocal;
311  }
312  rewards_(s, a) = experience_.getRewardSum(s, a) * visitSumReciprocal;
313  }

315  template <typename E>
316  void RLModel<E>::sync(const size_t s, const size_t a, const size_t s1) {
317  const auto visitSum = experience_.getVisitsSum(s, a);
318  // The second condition is related to numerical errors. Once in a
319  // while we reset those by forcing a true update using real data.
320  if ( !(visitSum % 10000ul) ) return sync(s, a);
321  if ( visitSum == 1ul ) {
322  transitions_[a].coeffRef(s, s) = 0.0;
323  transitions_[a].coeffRef(s, s1) = 1.0;
324  rewards_(s, a) = experience_.getReward(s, a, s1);
325  } else {
326  const double newVisits = static_cast<double>(experience_.getVisits(s, a, s1));
327 
328  // Update reward for this transition
329  rewards_(s, a) = experience_.getRewardSum(s, a) / visitSum;
330 
331  const double newTransitionValue = newVisits / static_cast<double>(visitSum - 1);
332  const double newVectorSum = 1.0 + (newTransitionValue - transitions_[a](s, s1));
333  // This works because as long as all the values in the transition have the same denominator
334  // (in this case visitSum-1), then the numerators do not matter, as we can simply normalize.
335  // In the end of the process the new values will be the same as if we updated directly using
336  // an increased denominator, and thus we will be able to call this function again correctly.
337  transitions_[a](s, s1) = newTransitionValue;
338  transitions_[a].row(s) /= newVectorSum;
339  }
340  }
