314  void RLModel<E>::sync(size_t s, size_t a) {
315  // Nothing to do
316  unsigned long visitSum = experience_.getVisitsSum(s, a);
317  if ( visitSum == 0ul ) return;
318 
319  // Create reciprocal for fast division
320  double visitSumReciprocal = 1.0 / visitSum;
321 
322  // Normalize
323  for ( size_t s1 = 0; s1 < S; ++s1 ) {
324  unsigned long visits = experience_.getVisits(s, a, s1);
325  // Normalize action reward over transition visits
326  if ( visits != 0 ) {
327  rewards_[a](s, s1) = experience_.getReward(s, a, s1) / visits;
328  }
329  transitions_[a](s, s1) = static_cast<double>(visits) * visitSumReciprocal;
330  }

333  template <typename E>
334  void RLModel<E>::sync(size_t s, size_t a, size_t s1) {
335  unsigned long visitSum = experience_.getVisitsSum(s, a);
336  // This function will not work on the first sync due to a division
337  // by zero (and possibly because the vector will not be empty, and
338  // even then then the computed new sum would be wrong). The second
339  // condition is related to numerical errors. Once in a while we reset
340  // those by forcing a true update using real data.
341  if ( visitSum < 2ul || !(visitSum % 10000ul) ) return sync(s, a);
342 
343  double newVisits = static_cast<double>(experience_.getVisits(s, a, s1));
344 
345  // Update reward for this transition (all others stay the same).
346  rewards_[a](s, s1) = experience_.getReward(s, a, s1) / newVisits;
347 
348  double newTransitionValue = newVisits / static_cast<double>(visitSum - 1);
349  double newVectorSum = 1.0 + (newTransitionValue - transitions_[a](s, s1));
350  // This works because as long as all the values in the transition have the same denominator
351  // (in this case visitSum-1), then the numerators do not matter, as we can simply normalize.
352  // In the end of the process the new values will be the same as if we updated directly using
353  // an increased denominator, and thus we will be able to call this function again correctly.
354  transitions_[a](s, s1) = newTransitionValue;
355  transitions_[a].row(s) /= newVectorSum;
356  }
