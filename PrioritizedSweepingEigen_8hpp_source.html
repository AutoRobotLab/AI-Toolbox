194  void PrioritizedSweepingEigen<M>::stepUpdateQ(const size_t s, const size_t a) {
195  // We use this to avoid continuous reallocations during the update
196  // of q[s][a]
197  static Values vector(S);
198 
199  auto & values = std::get<VALUES>(vfun_);
200  { // Update q[s][a]
201  vector.noalias() = values * model_.getDiscount();
202  vector += model_.getRewardFunction(a).row(s).transpose();
203  qfun_(s,a) = model_.getTransitionFunction(a).row(s).dot(vector);
204  }
205 
206  double p = values[s];
207  {
208  // Update value and action
209  values[s] = qfun_.row(s).maxCoeff(&std::get<ACTIONS>(vfun_)[s]);
210  }
211 
212  p = std::fabs(values[s] - p);
213 
214  // If it changed enough, we're going to update its parents.
215  if ( p > theta_ ) {
216  auto it = queueHandles_.find(s);
217 
218  if ( it != std::end(queueHandles_) && std::get<PRIORITY>(*(it->second)) < p )
219  queue_.increase(it->second, std::make_tuple(p, s));
220  else
221  queueHandles_[s] = queue_.push(std::make_tuple(p, s));
222  }
223  }

226  void PrioritizedSweepingEigen<M>::batchUpdateQ() {
227  for ( unsigned i = 0; i < N; ++i ) {
228  if ( queue_.empty() ) return;
229 
230  // The state we extract has been processed already
231  // So it is the future we have to backtrack from.
232  size_t s1;
233  std::tie(std::ignore, s1) = queue_.top();
234 
235  queue_.pop();
236  queueHandles_.erase(s1);
237 
238  for ( size_t s = 0; s < S; ++s )
239  for ( size_t a = 0; a < A; ++a )
240  if ( checkDifferentSmall(model_.getTransitionProbability(s,a,s1), 0.0) )
241  stepUpdateQ(s, a);
242  }
