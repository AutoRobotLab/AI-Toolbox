337  auto & obs = graph_.children[a].children;
338 
339  auto it = obs.find(o);
340  if ( it == obs.end() ) {
341  std::cerr << "Observation " << o << " never experienced in simulation, restarting with uniform belief..\n";
342  auto b = Belief(S); b.fill(1.0/S);
343  return sampleAction(b, horizon);

346  // Here we need an additional step, because *it is contained by graph_.
347  // If we just move assign, graph_ is first going to delete everything it

349  // into graph_! So we move *it outside of the graph_ hierarchy, so that

382  double POMCP<M>::simulate(BeliefNode & b, size_t s, unsigned depth) {
383  b.N++;
384 
385  auto begin = std::begin(b.children);
386  size_t a = std::distance(begin, findBestBonusA(begin, std::end(b.children), b.N));
387 
388  size_t s1, o; double rew;
389  std::tie(s1, o, rew) = model_.sampleSOR(s, a);
390 
391  auto & aNode = b.children[a];
392 
393  {
394  double futureRew = 0.0;
395  // We need to append the node anyway to perform the belief
396  // update for the next timestep.
397  auto ot = aNode.children.find(o);
398  if ( ot == std::end(aNode.children) ) {
399  aNode.children.emplace(std::piecewise_construct,
400  std::forward_as_tuple(o),
401  std::forward_as_tuple(s1));
402  // This stops automatically if we go out of depth
403  futureRew = rollout(s1, depth + 1);
404  }
405  else {
406  ot->second.belief.push_back(s1);
407  // We only go deeper if needed (maxDepth_ is always at least 1).
408  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) ) {
409  // Since most memory is allocated on the leaves,
410  // we do not allocate on node creation but only when
411  // we are actually descending into a node. If the node
412  // already has memory this should not do anything in
413  // any case.
414  ot->second.children.resize(A);
415  futureRew = simulate( ot->second, s1, depth + 1 );
416  }
417  }
418 
419  rew += model_.getDiscount() * futureRew;
420  }
421 
422  // Action update
423  aNode.N++;
424  aNode.V += ( rew - aNode.V ) / static_cast<double>(aNode.N);
425 
426  return rew;
427  }

451  Iterator POMCP<M>::findBestBonusA(Iterator begin, Iterator end, unsigned count) {
452  // Count here can be as low as 1.
453  // Since log(1) = 0, and 0/0 = error, we add 1.0.
454  double logCount = std::log(count + 1.0);
455  // We use this function to produce a score for each action. This can be easily
456  // substituted with something else to produce different POMCP variants.
457  auto evaluationFunction = [this, logCount](const ActionNode & an){
458  return an.V + exploration_ * std::sqrt( logCount / an.N );
459  };
460 
461  auto bestIterator = begin++;
462  double bestValue = evaluationFunction(*bestIterator);
463 
464  for ( ; begin < end; ++begin ) {
465  double actionValue = evaluationFunction(*begin);
466  if ( actionValue > bestValue ) {
467  bestValue = actionValue;
468  bestIterator = begin;
469  }
470  }
471 
472  return bestIterator;
473  }
