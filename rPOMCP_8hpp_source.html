189  if ( it == obs.end() ) {

227  template <typename M, bool UseEntropy>
228  double rPOMCP<M, UseEntropy>::simulate(BNode & b, size_t s, unsigned depth) {
229  b.N++;
230 
231  // Select next action node
232  auto begin = std::begin(b.children);
233  size_t a = std::distance(begin, findBestBonusA(begin, std::end(b.children), b.N));
234  auto & aNode = b.children[a];
235 
236  // Generate next step
237  size_t s1, o;
238  std::tie(s1, o, std::ignore) = model_.sampleSOR(s, a);
239 
240  double immAndFutureRew = 0.0;
241  {
242  typename decltype(aNode.children)::iterator ot;
243  bool newNode = false;
244 
245  // This either adds a node or sets ot to the existing node.
246  ot = aNode.children.find(o);
247  if ( ot == aNode.children.end() ) {
248  newNode = true;
249  std::tie(ot, std::ignore) = aNode.children.insert(std::make_pair(o, BNode()));
250  }
251 
252  // Compute knowledge for new observation node (entropy/max belief)
253  // This needs to be done here since we are going to upgrade a future belief.
254  ot->second.updateBeliefAndKnowledge(s1);
255 
256  // We only go deeper if needed (maxDepth_ is always at least 1).
257  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) && !newNode) {
258  ot->second.children.resize(A);
259  immAndFutureRew = simulate( ot->second, s1, depth + 1 );
260  }
261  // Otherwise we increase the N for the bottom leaves, since they can't get it otherwise and is needed for entropy
262  else {
263  ot->second.N += 1;
264  // For leaves we still extract entropy
265  if ( depth + 1 >= maxDepth_ )
266  immAndFutureRew = ot->second.getKnowledgeMeasure();
267  }
268  }
269 
270  // Action update
271  aNode.N += 1;
272  aNode.V += ( immAndFutureRew - aNode.V ) / static_cast<double>(aNode.N);
273 
274  // At this point the current beliefNode has a correct estimate of its
275  // own entropy. What it needs to do is select its best action. Although
276  // this is not needed for the top node.
277  if ( depth == 0 ) return 0.0;
278 
279  // Here we decide what to transmit to the upper level. In case this
280  // node has not been explored enough, then we simply pass on the new
281  // datapoint. Otherwise we compute the max over the actions, and we
282  // transmit a fake datapoint that will modify the value of the action
283  // above as if we chose the best action all the time in the past.
284  if ( b.N >= k_ ) {
285  // Force looking out for best action
286  if ( b.N == k_ ) {
287  b.actionsV = HUGE_VAL;
288  b.bestAction = a;
289  }
290  maxBeliefNodeUpdate(b, aNode, a);
291  }
292  else {
293  b.actionsV += ( immAndFutureRew - b.actionsV ) / static_cast<double>(b.N);
294  }
295 
296  double oldV = b.V;
297  // Note that both actionsV and entropy have been modified from last time!
298  // We discount the action part since it's the future reward part, while the
299  // immediate reward is the direct entropy, which is not discounted.
300  b.V = model_.getDiscount() * b.actionsV + b.getKnowledgeMeasure();
301  // This replaces our old value with the new value in the action update.
302  return (b.N - 1)*(b.V - oldV) + b.V;
303  }

327  template <typename Iterator>
328  Iterator rPOMCP<M, UseEntropy>::findBestBonusA(Iterator begin, Iterator end, unsigned count) {
329  // Count here can be as low as 1.
330  // Since log(1) = 0, and 0/0 = error, we add 1.0.
331  double logCount = std::log(count + 1.0);
332  // We use this function to produce a score for each action. This can be easily
333  // substituted with something else to produce different rPOMCP variants.
334  auto evaluationFunction = [this, logCount](const ANode & an){
335  return an.V + exploration_ * std::sqrt( logCount / an.N );
336  };
337 
338  auto bestIterator = begin++;
339  double bestValue = evaluationFunction(*bestIterator);
340 
341  for ( ; begin < end; ++begin ) {
342  double actionValue = evaluationFunction(*begin);
343  if ( actionValue > bestValue ) {
344  bestValue = actionValue;
345  bestIterator = begin;
346  }
347  }
348 
349  return bestIterator;
350  }
