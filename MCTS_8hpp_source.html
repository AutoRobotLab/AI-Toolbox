196  size_t MCTS<M>::sampleAction(size_t a, size_t s1, unsigned horizon) {
197  auto & states = graph_.children[a].children;
198 
199  auto it = states.find(s1);
200  if ( it == states.end() )
201  return sampleAction(s1, horizon);
202 
203  // Here we need an additional step, because *it is contained by graph_.
204  // If we just move assign, graph_ is first going to delete everything it
205  // contains (included *it), and then we are going to move unallocated memory
206  // into graph_! So we move *it outside of the graph_ hierarchy, so that
207  // we can then assign safely.
208  { auto tmp = std::move(it->second); graph_ = std::move(tmp); }
209 
210  return runSimulation(s1, horizon);
211  }

226  template <typename M>
227  double MCTS<M>::simulate(StateNode & sn, size_t s, unsigned depth) {
228  // Head update
229  sn.N++;
230 
231  auto begin = std::begin(sn.children);
232  size_t a = std::distance(begin, findBestBonusA(begin, std::end(sn.children), sn.N));
233 
234  size_t s1; double rew;
235  std::tie(s1, rew) = model_.sampleSR(s, a);
236 
237  auto & aNode = sn.children[a];
238 
239  // We only go deeper if needed (maxDepth_ is always at least 1).
240  if ( depth + 1 < maxDepth_ && !model_.isTerminal(s1) ) {
241  auto end = std::end(aNode.children);
242  auto it = aNode.children.find(s1);
243 
244  double futureRew;
245  if ( it == end ) {
246  aNode.children.emplace(std::piecewise_construct,
247  std::forward_as_tuple(s1),
248  std::forward_as_tuple(A));
249  futureRew = rollout(s1, depth + 1);
250  }
251  else {
252  futureRew = simulate( it->second, s1, depth + 1 );
253  }
254 
255  rew += model_.getDiscount() * futureRew;
256  }
257 
258  // Action update
259  aNode.N++;
260  aNode.V += ( rew - aNode.V ) / static_cast<double>(aNode.N);
261 
262  return rew;
263  }

266  double MCTS<M>::rollout(size_t s, unsigned depth) {
267  double rew = 0.0, totalRew = 0.0, gamma = 1.0;
268 
269  std::uniform_int_distribution<size_t> generator(0, A-1);
270  for ( ; depth < maxDepth_; ++depth ) {
271  std::tie( s, rew ) = model_.sampleSR( s, generator(rand_) );
272 
273  totalRew += gamma * rew;
274  gamma *= model_.getDiscount();
275  }

286  template <typename Iterator>
287  Iterator MCTS<M>::findBestBonusA(Iterator begin, Iterator end, unsigned count) {
288  // Count here can be as low as 1.
289  // Since log(1) = 0, and 0/0 = error, we add 1.0.
290  double logCount = std::log(count + 1.0);
291  // We use this function to produce a score for each action. This can be easily
292  // substituted with something else to produce different POMCP variants.
293  auto evaluationFunction = [this, logCount](const ActionNode & an){
294  return an.V + exploration_ * std::sqrt( logCount / an.N );
295  };
296 
297  auto bestIterator = begin++;
298  double bestValue = evaluationFunction(*bestIterator);
299 
300  for ( ; begin < end; ++begin ) {
301  double actionValue = evaluationFunction(*begin);
302  if ( actionValue > bestValue ) {
303  bestValue = actionValue;
304  bestIterator = begin;
305  }
306  }
307 
308  return bestIterator;
309  }
