321  template <typename E>
322  void SparseRLModel<E>::sync(const size_t s, const size_t a) {
323  // Nothing to do
324  const auto visitSum = experience_.getVisitsSum(s, a);
325  if ( visitSum == 0ul ) return;
326 
327  // Create reciprocal for fast division
328  const double visitSumReciprocal = 1.0 / visitSum;
329 
330  // Normalize
331  for ( size_t s1 = 0; s1 < S; ++s1 ) {
332  const auto visits = experience_.getVisits(s, a, s1);
333  // Normalize action reward over transition visits
334  if ( visits != 0 ) {
335  rewards_[a].coeffRef(s, s1) = experience_.getReward(s, a, s1) / visits;
336  }
337  transitions_[a].coeffRef(s, s1) = static_cast<double>(visits) * visitSumReciprocal;
338  }
339  }

341  template <typename E>
342  void SparseRLModel<E>::sync(const size_t s, const size_t a, const size_t s1) {
343  const auto visitSum = experience_.getVisitsSum(s, a);
344  // The second condition is related to numerical errors. Once in a
345  // while we reset those by forcing a true update using real data.
346  if ( !(visitSum % 10000ul) ) return sync(s, a);
347  if ( visitSum == 1ul ) {
348  transitions_[a].coeffRef(s, s) = 0.0;
349  transitions_[a].coeffRef(s, s1) = 1.0;
350  rewards_[a].coeffRef(s, s1) = experience_.getReward(s, a, s1);
351  } else {
352  const double newVisits = static_cast<double>(experience_.getVisits(s, a, s1));
353 
354  // Update reward for this transition (all others stay the same).
355  rewards_[a].coeffRef(s, s1) = experience_.getReward(s, a, s1) / newVisits;
356 
357  const double newTransitionValue = newVisits / static_cast<double>(visitSum - 1);
358  const double newVectorSum = 1.0 + (newTransitionValue - transitions_[a].coeff(s, s1));
359  // This works because as long as all the values in the transition have the same denominator
360  // (in this case visitSum-1), then the numerators do not matter, as we can simply normalize.
361  // In the end of the process the new values will be the same as if we updated directly using
362  // an increased denominator, and thus we will be able to call this function again correctly.
363  transitions_[a].coeffRef(s, s1) = newTransitionValue;
364  transitions_[a].row(s) /= newVectorSum;
365  }
