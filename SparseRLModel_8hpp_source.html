310  template <typename E>
311  void SparseRLModel<E>::sync(const size_t s, const size_t a) {
312  // Nothing to do
313  const auto visitSum = experience_.getVisitsSum(s, a);
314  if ( visitSum == 0ul ) return;
315  // Clear beginning's identity matrix
316  if ( visitSum == 1ul )
317  transitions_[a].coeffRef(s, s) = 0.0;
318 
319  // Create reciprocal for fast division
320  const double visitSumReciprocal = 1.0 / visitSum;
321 
322  // Normalize
323  for ( size_t s1 = 0; s1 < S; ++s1 ) {
324  const auto visits = experience_.getVisits(s, a, s1);
325  if (visits > 0)
326  transitions_[a].coeffRef(s, s1) = static_cast<double>(visits) * visitSumReciprocal;
327  }
328  if (checkDifferentSmall(0.0, experience_.getRewardSum(s, a)))
329  rewards_.coeffRef(s, a) = experience_.getRewardSum(s, a) * visitSumReciprocal;
330  }

332  template <typename E>
333  void SparseRLModel<E>::sync(const size_t s, const size_t a, const size_t s1) {
334  const auto visitSum = experience_.getVisitsSum(s, a);
335  // The second condition is related to numerical errors. Once in a
336  // while we reset those by forcing a true update using real data.
337  if ( !(visitSum % 10000ul) ) return sync(s, a);
338  if ( visitSum == 1ul ) {
339  transitions_[a].coeffRef(s, s) = 0.0;
340  transitions_[a].coeffRef(s, s1) = 1.0;
341  if (checkDifferentSmall(0.0, experience_.getRewardSum(s, a)))
342  rewards_.coeffRef(s, a) = experience_.getRewardSum(s, a) / visitSum;
343  } else {
344  const double newVisits = static_cast<double>(experience_.getVisits(s, a, s1));
345 
346  if (checkDifferentSmall(0.0, experience_.getRewardSum(s, a)))
347  rewards_.coeffRef(s, a) = experience_.getRewardSum(s, a) / visitSum;
348 
349  const double newTransitionValue = newVisits / static_cast<double>(visitSum - 1);
350  const double newVectorSum = 1.0 + (newTransitionValue - transitions_[a].coeff(s, s1));
351  // This works because as long as all the values in the transition have the same denominator
352  // (in this case visitSum-1), then the numerators do not matter, as we can simply normalize.
353  // In the end of the process the new values will be the same as if we updated directly using
354  // an increased denominator, and thus we will be able to call this function again correctly.
355  transitions_[a].coeffRef(s, s1) = newTransitionValue;
356  transitions_[a].row(s) /= newVectorSum;
357  }
