103  if ( !bl ) return;

105 
106  // Since the original method of obtaining beliefs is stochastic,
107  // we keep trying for a while in case we don't find any new beliefs.
108  // However, for some problems (for example the Tiger problem) still
109  // this does not perform too well since the probability of finding
110  // a new belief (via action LISTEN) is pretty low.
111  size_t currentSize = beliefs.size();
112  size_t firstProductiveBelief = 0;
113  size_t bonusBeliefsToAdd = std::max((size_t)1, beliefNumber / 20);
114  while ( currentSize < beliefNumber ) {

118  if ( currentSize == beliefs.size() ) ++counter;
119  else {

142  return (lhs - rhs).cwiseAbs().sum();

146  // We apply the discovery process also to all beliefs we discover
147  // along the way. We start from the first good one, since the others
148  // have already produced as much as they can.
149  for (size_t i = firstProductiveBelief; i < bl.size(); ++i) {
150  // Compute all new beliefs
151  for ( size_t a = 0; a < A; ++a ) {

161  // elements, or vs what we have already produced in this round
162  // of 20.
163  if (checkDifferentSmall(distances[a], 0.0))
164  distance = computeDistance(helper, newBeliefs[a]);
165  else
