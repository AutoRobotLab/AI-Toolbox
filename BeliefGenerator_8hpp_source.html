102  void BeliefGenerator<M>::operator()(const size_t beliefNumber, BeliefList * bl) const {
103  if ( !bl ) return;
104  auto & beliefs = *bl;
105 
106  // Since the original method of obtaining beliefs is stochastic,
107  // we keep trying for a while in case we don't find any new beliefs.
108  // However, for some problems (for example the Tiger problem) still
109  // this does not perform too well since the probability of finding
110  // a new belief (via action LISTEN) is pretty low.
111  size_t currentSize = beliefs.size();
112  size_t firstProductiveBelief = 0;
113  size_t bonusBeliefsToAdd = std::max((size_t)1, beliefNumber / 20);
114  while ( currentSize < beliefNumber ) {
115  unsigned counter = 0;
116  while ( counter < 5 ) {
117  expandBeliefList(beliefNumber, firstProductiveBelief, &beliefs);
118  if ( currentSize == beliefs.size() ) ++counter;
119  else {
120  counter = 0;
121  currentSize = beliefs.size();
122  if ( currentSize >= beliefNumber ) return;
123  }
124  }
125  firstProductiveBelief = beliefs.size();
126  for ( size_t i = 0; currentSize < beliefNumber && i < bonusBeliefsToAdd; ++i, ++currentSize )
127  beliefs.emplace_back(makeRandomProbability(S, rand_));

147  // We apply the discovery process also to all beliefs we discover
148  // along the way. We start from the first good one, since the others
149  // have already produced as much as they can.
150  for (size_t i = firstProductiveBelief; i < bl.size(); ++i) {
151  // Compute all new beliefs
152  for ( size_t a = 0; a < A; ++a ) {

164  // produced this round. If it passes, add it to them.
165  bool pass = true;
166  for ( unsigned k = 0; k < bufferFill; ++k ) {
167  if (o == observationBuffer[k]) {
168  pass = false;
169  break;
170  }
171  }
172  if (!pass) continue;

175  observationBuffer[bufferFill++] = o;

179  for (size_t k = 0; k < bl.size(); ++k) {

196  bl.emplace_back(std::move(newBeliefs[id]));

198  }
