171  void PrioritizedSweeping<M>::stepUpdateQ(size_t s, size_t a) {
172  { // Update q[s][a]
173  double newQValue = 0;
174  for ( size_t s1 = 0; s1 < S; ++s1 ) {
175  double probability = model_.getTransitionProbability(s,a,s1);
176  if ( probability > 0.0 )
177  newQValue += probability * ( model_.getExpectedReward(s,a,s1) + model_.getDiscount() * vfun_[s1] );
178  }
179  qfun_[s][a] = newQValue;
180  }
181 
182  double p = vfun_[s];
183  vfun_[s] = *std::max_element(std::begin(qfun_[s]), std::end(qfun_[s]));
184 
185  p = std::fabs(vfun_[s] - p);
186 
187  // If it changed enough, we're going to update its parents.
188  if ( p > theta_ ) {
189  auto it = queueHandles_.find(s);
190 
191  if ( it != std::end(queueHandles_) && std::get<0>(*(it->second)) < p )
192  queue_.increase(it->second, std::make_tuple(p, s));
193  else
194  queueHandles_[s] = queue_.push(std::make_tuple(p, s));
195  }
196  }

199  void PrioritizedSweeping<M>::batchUpdateQ() {
200  for ( unsigned i = 0; i < N; ++i ) {
201  if ( queue_.empty() ) return;
202 
203  // The state we extract has been processed already
204  // So it is the future we have to backtrack from.
