198  void PrioritizedSweeping<M>::stepUpdateQ(size_t s, size_t a) {
199  auto & values = std::get<VALUES>(vfun_);
200  { // Update q[s][a]
201  double newQValue = 0;
202  for ( size_t s1 = 0; s1 < S; ++s1 ) {
203  double probability = model_.getTransitionProbability(s,a,s1);
204  if ( checkDifferentSmall( probability, 0.0 ) )
205  newQValue += probability * ( model_.getExpectedReward(s,a,s1) + model_.getDiscount() * values[s1] );
206  }
207  qfun_[s][a] = newQValue;
208  }
209 
210  double p = values[s];
211  {
212  auto ref = qfun_[s];
213  auto begin = std::begin(ref);
214  auto it = std::max_element(begin, std::end(ref));
215  // Update value and action
216  values[s] = *it;
217  std::get<ACTIONS>(vfun_)[s] = std::distance(begin, it);
218  }
219 
220  p = std::fabs(values[s] - p);
221 
222  // If it changed enough, we're going to update its parents.
223  if ( p > theta_ ) {
224  auto it = queueHandles_.find(s);
225 
226  if ( it != std::end(queueHandles_) && std::get<PRIORITY>(*(it->second)) < p )
227  queue_.increase(it->second, std::make_tuple(p, s));
228  else
229  queueHandles_[s] = queue_.push(std::make_tuple(p, s));
230  }
231  }

234  void PrioritizedSweeping<M>::batchUpdateQ() {
235  for ( unsigned i = 0; i < N; ++i ) {
236  if ( queue_.empty() ) return;
237 
238  // The state we extract has been processed already
239  // So it is the future we have to backtrack from.
240  size_t s1;
241  std::tie(std::ignore, s1) = queue_.top();
242 
243  queue_.pop();
244  queueHandles_.erase(s1);
245 
246  for ( size_t s = 0; s < S; ++s )
247  for ( size_t a = 0; a < A; ++a )
248  if ( checkDifferentSmall(model_.getTransitionProbability(s,a,s1), 0.0) )
249  stepUpdateQ(s, a);
250  }
