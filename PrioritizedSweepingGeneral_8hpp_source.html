197  void PrioritizedSweepingGeneral<M>::stepUpdateQ(const size_t s, const size_t a) {
198  auto & values = std::get<VALUES>(vfun_);
199  { // Update q[s][a]
200  double newQValue = 0;
201  for ( size_t s1 = 0; s1 < S; ++s1 ) {
202  const double probability = model_.getTransitionProbability(s,a,s1);
203  if ( checkDifferentSmall( probability, 0.0 ) )
204  newQValue += probability * ( model_.getExpectedReward(s,a,s1) + model_.getDiscount() * values[s1] );
205  }
206  qfun_(s, a) = newQValue;
207  }
208 
209  double p = values[s];
210  {
211  // Update value and action
212  values[s] = qfun_.row(s).maxCoeff(&std::get<ACTIONS>(vfun_)[s]);
213  }
214 
215  p = std::fabs(values[s] - p);
216 
217  // If it changed enough, we're going to update its parents.
218  if ( p > theta_ ) {
219  auto it = queueHandles_.find(s);
220 
221  if ( it != std::end(queueHandles_) && std::get<PRIORITY>(*(it->second)) < p )
222  queue_.increase(it->second, std::make_tuple(p, s));
223  else
224  queueHandles_[s] = queue_.push(std::make_tuple(p, s));
225  }
226  }

229  void PrioritizedSweepingGeneral<M>::batchUpdateQ() {
230  for ( unsigned i = 0; i < N; ++i ) {
231  if ( queue_.empty() ) return;
232 
233  // The state we extract has been processed already
234  // So it is the future we have to backtrack from.
235  size_t s1;
236  std::tie(std::ignore, s1) = queue_.top();
237 
238  queue_.pop();
239  queueHandles_.erase(s1);
240 
241  for ( size_t s = 0; s < S; ++s )
242  for ( size_t a = 0; a < A; ++a )
243  if ( checkDifferentSmall(model_.getTransitionProbability(s,a,s1), 0.0) )
244  stepUpdateQ(s, a);
245  }
246  }
