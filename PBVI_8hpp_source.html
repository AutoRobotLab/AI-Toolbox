167  std::tuple<bool, ValueFunction> PBVI::operator()(const M & model) {
168  // Initialize "global" variables
169  S = model.getS();
170  A = model.getA();
171  O = model.getO();
172 
173  // In this implementation we compute all beliefs in advance. This
174  // is mostly due to the fact that I prefer counter parameters (how
175  // many beliefs do you want?) versus timers (loop until time is
176  // up). However, this is easily changeable, since the belief generator
177  // can be called multiple times to increase the size of the belief
178  // vector.
179  BeliefGenerator<M> bGen(model);
180  auto beliefs = bGen(beliefSize_);
181 
182  ValueFunction v(1, VList(1, makeVEntry(S)));
183 
184  unsigned timestep = 0;
185 
186  Projecter<M> projecter(model);
187 
188  // And off we go
189  bool useEpsilon = checkDifferentSmall(epsilon_, 0.0);
190  double variation = epsilon_ * 2; // Make it bigger
191  while ( timestep < horizon_ && ( !useEpsilon || variation > epsilon_ ) ) {
192  ++timestep;
193 
194  // Compute all possible outcomes, from our previous results.
195  // This means that for each action-observation pair, we are going
196  // to obtain the same number of possible outcomes as the number
197  // of entries in our initial vector w.
198  auto projs = projecter(v[timestep-1]);
199 
200  size_t finalWSize = 0;
201  // In this method we split the work by action, which will then
202  // be joined again at the end of the loop. This is not required,
203  // but there does not seem to be a speed boost by not doing
204  // so (not that I found one, if there is one I'd like to know!)
205  for ( size_t a = 0; a < A; ++a ) {
206  projs[a][0] = crossSum( projs[a], a, beliefs );
207  finalWSize += projs[a][0].size();
208  }
209  VList w;
210  w.reserve(finalWSize);
211 
212  for ( size_t a = 0; a < A; ++a )
213  std::move(std::begin(projs[a][0]), std::end(projs[a][0]), std::back_inserter(w));
214 
215  auto begin = std::begin(w), bound = begin, end = std::end(w);
216  for ( auto & belief : beliefs )
217  bound = extractWorstAtBelief(belief, begin, bound, end);
218 
219  w.erase(bound, end);
220 
221  // If you want to save as much memory as possible, do this.
222  // It make take some time more though since it needs to reallocate
223  // and copy stuff around.
224  // w.shrink_to_fit();
225 
226  v.emplace_back(std::move(w));
227 
228  // Check convergence
229  if ( useEpsilon ) {
230  variation = weakBoundDistance(v[timestep-1], v[timestep]);
231  }
232  }
233 
234  return std::make_tuple(true, v);
235  }
