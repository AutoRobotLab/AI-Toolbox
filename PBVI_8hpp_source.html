140  std::tuple<bool, ValueFunction> PBVI::operator()(const M & model) {
141  // Initialize "global" variables
142  S = model.getS();
143  A = model.getA();
144  O = model.getO();
145 
146  // In this implementation we compute all beliefs in advance. This
147  // is mostly due to the fact that I prefer counter parameters (how
148  // many beliefs do you want?) versus timers (loop until time is
149  // up). However, this is easily changeable, since the belief generator
150  // can be called multiple times to increase the size of the belief
151  // vector.
152  BeliefGenerator<M> bGen(model);
153  auto beliefs = bGen(beliefSize_);
154 
155  ValueFunction v(1, VList(1, makeVEntry(S)));
156 
157  unsigned timestep = 1;
158 
159  Projecter<M> projecter(model);
160 
161  // And off we go
162  while ( timestep <= horizon_ ) {
163  // Compute all possible outcomes, from our previous results.
164  // This means that for each action-observation pair, we are going
165  // to obtain the same number of possible outcomes as the number
166  // of entries in our initial vector w.
167  auto projs = projecter(v[timestep-1]);
168 
169  size_t finalWSize = 0;
170  // In this method we split the work by action, which will then
171  // be joined again at the end of the loop.
172  for ( size_t a = 0; a < A; ++a ) {
173  projs[a][0] = crossSum( projs[a], a, beliefs );
174  finalWSize += projs[a][0].size();
175  }
176  VList w;
177  w.reserve(finalWSize);
178 
179  for ( size_t a = 0; a < A; ++a )
180  std::move(std::begin(projs[a][0]), std::end(projs[a][0]), std::back_inserter(w));
181 
182  auto begin = std::begin(w), bound = begin, end = std::end(w);
183  for ( auto & belief : beliefs )
184  bound = extractWorstAtBelief(std::begin(belief), std::end(belief), begin, bound, end);
185 
186  w.erase(bound, end);
187 
188  // If you want to save as much memory as possible, do this.
189  // It make take some time more though since it needs to reallocate
190  // and copy stuff around.
191  // w.shrink_to_fit();
192 
193  v.emplace_back(std::move(w));
194 
195  ++timestep;
196  }
197 
198  return std::make_tuple(true, v);
199  }

203  VList result;

206  for ( auto & b : bl ) {

211  for ( size_t o = 0; o < O; ++o ) {
212  const VList & projsO = projs[o];
