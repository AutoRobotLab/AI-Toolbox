144  std::tuple<bool, ValueFunction> PBVI::operator()(const M & model) {
145  // Initialize "global" variables
146  S = model.getS();
147  A = model.getA();
148  O = model.getO();
149 
150  // In this implementation we compute all beliefs in advance. This
151  // is mostly due to the fact that I prefer counter parameters (how
152  // many beliefs do you want?) versus timers (loop until time is
153  // up). However, this is easily changeable, since the function that
154  // computes beliefs itself respects the interface defined in the
155  // original PBVI paper (it tries to double the belief list given to
156  // it).
157  BeliefList beliefs(1, Belief(S, 0.0)); beliefs[0][0] = 1.0; // TODO: May take user input
158 
159  // Since the original method of obtaining beliefs is stochastic,
160  // we keep trying for a while in case we don't find any new beliefs.
161  // However, for some problems (for example the Tiger problem) still
162  // this does not perform too well since the probability of finding
163  // a new belief (via action LISTEN) is pretty low.
164  size_t currentSize = beliefs.size(); unsigned counter = 0;
165  while ( currentSize < beliefSize_ && counter < 30 ) {
166  expandBeliefs(model, beliefs);
167  if ( currentSize == beliefs.size() ) ++counter;
168  currentSize = beliefs.size();
169  }
170 
171  ValueFunction v(1, VList(1, makeVEntry(S)));
172 
173  unsigned timestep = 1;
174 
175  Projecter<M> projecter(model);
176 
177  // And off we go
178  while ( timestep <= horizon_ ) {
179  // Compute all possible outcomes, from our previous results.
180  // This means that for each action-observation pair, we are going
181  // to obtain the same number of possible outcomes as the number
182  // of entries in our initial vector w.
183  auto projs = projecter(v[timestep-1]);
184 
185  size_t finalWSize = 0;
186  // In this method we split the work by action, which will then
187  // be joined again at the end of the loop.
188  for ( size_t a = 0; a < model.getA(); ++a ) {
189  projs[a][0] = crossSum( projs[a], a, beliefs );
190  finalWSize += projs[a][0].size();
191  }
192  VList w;
193  w.reserve(finalWSize);
194 
195  for ( size_t a = 0; a < model.getA(); ++a )
196  std::move(std::begin(projs[a][0]), std::end(projs[a][0]), std::back_inserter(w));
197 
198  v.emplace_back(std::move(w));
199 
200  ++timestep;
201  }
202 
203  return std::make_tuple(true, v);
204  }

207  void PBVI::expandBeliefs(const M& model, BeliefList & bl) const {
208  BeliefList newBeliefs(A);
209  std::vector<double> distances(A);
210  auto dBegin = std::begin(distances), dEnd = std::end(distances);
211 
212  size_t size = bl.size();
213  bl.reserve(size * 2);
214 
215  // L1 distance
216  auto computeDistance = [](const Belief & lhs, const Belief & rhs, size_t S) {
217  double distance = 0.0;
218  for ( size_t i = 0; i < S; ++i )
219  distance += std::abs(lhs[i] - rhs[i]);
220  return distance;
221  };
222 
223  for ( size_t i = 0; i < size; ++i ) {
224  // Compute all new beliefs
225  for ( size_t a = 0; a < A; ++a ) {
226  size_t s = sampleProbability(S, bl[i], rand_);
227 
228  size_t o;
229  std::tie(std::ignore, o, std::ignore) = model.sampleSOR(s, a);
230  newBeliefs[a] = updateBelief(model, bl[i], a, o);
231 
232  // Compute distance (here we compare also against elements we just added!)
233  distances[a] = computeDistance(newBeliefs[a], bl[0], S);
234  for ( size_t j = 1; j < bl.size(); ++j ) {
235  if ( checkEqual(distances[a], 0.0) ) break; // We already have it!
236  distances[a] = std::min(distances[a], computeDistance(newBeliefs[a], bl[j], S));
237  }
238  }
239  // Find furthest away, add only if it is new.
240  size_t id = std::distance( dBegin, std::max_element(dBegin, dEnd) );
241  if ( checkDifferent(distances[id], 0.0) )
242  bl.emplace_back(std::move(newBeliefs[id]));
243  }
244  }

247  VList PBVI::crossSum(const ProjectionsRow & projs, size_t a, const BeliefList & bl) {
248  VList result;
249  result.reserve(bl.size());
250 
251  for ( auto & b : bl ) {
252  MDP::Values v(S, 0.0);
253  VObs obs(O, 0);
254 
255  // We compute the crossSum between each best vector for the belief.
256  for ( size_t o = 0; o < O; ++o ) {
257  const VList & projsO = projs[o];
258  auto bestMatch = findBestAtBelief(S, b, std::begin(projsO), std::end(projsO));
259 
260  for ( size_t s = 0; s < S; ++s )
261  v[s] += std::get<VALUES>(*bestMatch)[s];
262 
263  obs[o] = std::get<OBS>(*bestMatch)[0];
264  }
265  result.emplace_back(v, a, obs);
266  }
267  // We could probably do better than this but what the hell.
268  Pruner::dominationPrune( S, &result );
