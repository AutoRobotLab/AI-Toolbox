206  std::tuple<double, ValueFunction> PBVI::operator()(const M & model, const std::vector<Belief> & beliefs, ValueFunction v) {
207  // Initialize "global" variables
208  S = model.getS();
209  A = model.getA();
210  O = model.getO();
211 
212  if (v.size() == 0)
213  v = makeValueFunction(S);
214 
215  unsigned timestep = 0;
216 
217  Projecter projecter(model);
218 
219  // And off we go
220  const bool useEpsilon = checkDifferentSmall(epsilon_, 0.0);
221  double variation = epsilon_ * 2; // Make it bigger
222  while ( timestep < horizon_ && ( !useEpsilon || variation > epsilon_ ) ) {
223  ++timestep;
224 
225  // Compute all possible outcomes, from our previous results.
226  // This means that for each action-observation pair, we are going
227  // to obtain the same number of possible outcomes as the number
228  // of entries in our initial vector w.
229  auto projs = projecter(v.back());
230 
231  size_t finalWSize = 0;
232  // In this method we split the work by action, which will then
233  // be joined again at the end of the loop. This is not required,
234  // but there does not seem to be a speed boost by not doing
235  // so (not that I found one, if there is one I'd like to know!)
236  for ( size_t a = 0; a < A; ++a ) {
237  projs[a][0] = crossSum( projs[a], a, beliefs );
238  finalWSize += projs[a][0].size();
239  }
240  VList w;
241  w.reserve(finalWSize);
242 
243  for ( size_t a = 0; a < A; ++a )
244  std::move(std::begin(projs[a][0]), std::end(projs[a][0]), std::back_inserter(w));
245 
246  auto begin = std::begin(w), bound = begin, end = std::end(w);
247  for ( const auto & belief : beliefs )
248  bound = extractBestAtBelief(belief, begin, bound, end);
249 
250  w.erase(bound, end);
251 
252  // If you want to save as much memory as possible, do this.
253  // It make take some time more though since it needs to reallocate
254  // and copy stuff around.
255  // w.shrink_to_fit();
256 
257  v.emplace_back(std::move(w));
258 
259  // Check convergence
260  if ( useEpsilon )
261  variation = weakBoundDistance(v[v.size()-2], v.back());
262  }
263 
264  return std::make_tuple(useEpsilon ? variation : 0.0, v);
265  }
