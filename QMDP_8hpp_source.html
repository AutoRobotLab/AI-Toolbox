134  VList w;

137  // We simply create a VList where each entry is a slice of the MDP::ValueFunction.
138  // All in all, since an MDP is a subclass of POMDPs, what we are doing is writing the
139  // true representation of the MDP solution (where we know only the solutions for the
140  // corners of the belief space). The MDP::ValueFunction is simply a condensed form of
141  // the solution, since in an MDP the only "beliefs" we have are the corners.
142  for ( size_t s = 0; s < S; ++s ) {
143  MDP::Values v(S); v.fill(0.0);
144  v[s] = mdpValues[s];
145  // All observations are 0 since we go back to the horizon 0 entry, which is nil.
146  w.emplace_back(v, mdpActions[s], VObs(m.getO(), 0u));
147  }

149  ValueFunction vf(1, VList(1, makeVEntry(S)));
150  vf.emplace_back(std::move(w));
151
