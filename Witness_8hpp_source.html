172  unsigned timestep = 0;
173 
174  // This variable we use to manually control the allocations
175  // for the LP solver. This is because this algorithm cannot
176  // know in advance just how many constraints the LP is going
177  // to get. Thus we implement a x2 doubling allocation scheme
178  // to avoid too many reallocations.
179  size_t reserveSize = 1;
180 
181  Projecter project(model);
182  Pruner<WitnessLP> prune(S);

184 
185  const bool useEpsilon = checkDifferentSmall(epsilon_, 0.0);
186  double variation = epsilon_ * 2; // Make it bigger
187  while ( timestep < horizon_ && ( !useEpsilon || variation > epsilon_ ) ) {
188  ++timestep;
189 
190  // As default, we allocate double the numbers of VEntries for last step.
191  reserveSize = std::max(reserveSize, 2 * v[timestep-1].size());
192  // Compute all possible outcomes, from our previous results.
193  // This means that for each action-observation pair, we are going
194  // to obtain the same number of possible outcomes as the number
195  // of entries in our initial vector w.
196  auto projections = project(v[timestep-1]);
197 
198  size_t finalWSize = 0;
199  for ( size_t a = 0; a < A; ++a ) {
200  U[a].clear();
201  lp.reset();
202  agenda_.clear();
203  triedVectors_.clear();
204  size_t counter = 0;
205 
206  lp.allocate(reserveSize);
207 
208  // We add the VEntry to startoff the whole process. This
209  // VEntry does not even need to be optimal, as we are going
210  // to compute the optimal one for the witness point anyway.
211  addDefaultEntry(projections[a], a);
212 
213  // We check whether any element in the agenda improves what we have
214  while ( !agenda_.empty() ) {
215  const auto witness = lp.findWitness( std::get<VALUES>(agenda_.back()) );
216  if ( witness ) {
217  // If so, we generate the best vector for that particular belief point.
218  U[a].push_back(crossSumBestAtBelief(projections[a], a, *witness));
219  lp.addOptimalRow(std::get<VALUES>(U[a].back()));
220  // We add to the agenda all possible "variations" of the VEntry found.
221  addVariations(projections[a], a, U[a].back());
222  // We manually check memory for the lp, since this method
223  // cannot know in advance how many rows it'll need to do.
224  if ( ++counter == reserveSize ) {
225  reserveSize *= 2;
226  lp.allocate(reserveSize);
227  }
228  }
229  else
230  agenda_.pop_back();
231  }
232  finalWSize += U[a].size();
233  }

236

238  for ( size_t a = 0; a < A; ++a )
239  std::move(std::begin(U[a]), std::end(U[a]), std::back_inserter(w));
240

242  // computed the parsimonious set of value functions.
243  prune( &w );
244  v.emplace_back(std::move(w));
245 
246  // Check convergence
247  if ( useEpsilon ) {
248  variation = weakBoundDistance(v[timestep-1], v[timestep]);
249  }
250  }
