135  std::tuple<bool, ValueFunction> PERSEUS::operator()(const M & model, double minReward) {
136  if ( model.getDiscount() == 1 ) throw std::invalid_argument("The model cannot have a discount of 1 in PERSEUS!");
137  // Initialize "global" variables
138  S = model.getS();
139  A = model.getA();
140  O = model.getO();
141 
142  // In this implementation we compute all beliefs in advance. This
143  // is mostly due to the fact that I prefer counter parameters (how
144  // many beliefs do you want?) versus timers (loop until time is
145  // up). However, this is easily changeable, since the belief generator
146  // can be called multiple times to increase the size of the belief
147  // vector.
148  BeliefGenerator<M> bGen(model);
149  auto beliefs = bGen(beliefSize_);
150 
151  // We initialize the ValueFunction to the "worst" case scenario.
152  ValueFunction v(1, VList(1, std::make_tuple(MDP::Values(S, minReward / (1.0 - model.getDiscount())), 0, VObs(0))));
153 
154  unsigned timestep = 0;
155 
156  Projecter<M> projecter(model);
157 
158  // And off we go
159  bool useEpsilon = checkDifferentSmall(epsilon_, 0.0);
160  double variation = epsilon_ * 2; // Make it bigger
161  while ( timestep < horizon_ && ( !useEpsilon || variation > epsilon_ ) ) {
162  ++timestep;
163  // Compute all possible outcomes, from our previous results.
164  // This means that for each action-observation pair, we are going
165  // to obtain the same number of possible outcomes as the number
166  // of entries in our initial vector w.
167  auto projs = projecter(v[timestep-1]);
168  // Here we find the minimum number of VEntries that we need to improve
169  // v on all beliefs from v[timestep-1].
170  v.emplace_back( crossSum( projs, beliefs, v[timestep-1] ) );
171 
172  // Check convergence
173  if ( useEpsilon ) {
174  variation = weakBoundDistance(v[timestep-1], v[timestep]);
175  }
176  }
177 
178  return std::make_tuple(true, v);
179  }
