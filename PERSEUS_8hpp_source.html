159  std::tuple<bool, ValueFunction> PERSEUS::operator()(const M & model, const double minReward) {
160  if ( model.getDiscount() == 1 ) throw std::invalid_argument("The model cannot have a discount of 1 in PERSEUS!");
161  // Initialize "global" variables
162  S = model.getS();
163  A = model.getA();
164  O = model.getO();
165 
166  // In this implementation we compute all beliefs in advance. This
167  // is mostly due to the fact that I prefer counter parameters (how
168  // many beliefs do you want?) versus timers (loop until time is
169  // up). However, this is easily changeable, since the belief generator
170  // can be called multiple times to increase the size of the belief
171  // vector.
172  BeliefGenerator bGen(model);
173  const auto beliefs = bGen(beliefSize_);
174 
175  // We initialize the ValueFunction to the "worst" case scenario.
176  ValueFunction v(1, VList(1, std::make_tuple(MDP::Values(S), 0, VObs(0))));
177  std::get<VALUES>(v[0][0]).fill(minReward / (1.0 - model.getDiscount()));
178 
179  unsigned timestep = 0;
180 
181  Projecter projecter(model);
182 
183  // And off we go
184  const bool useEpsilon = checkDifferentSmall(epsilon_, 0.0);
185  double variation = epsilon_ * 2; // Make it bigger
186  while ( timestep < horizon_ && ( !useEpsilon || variation > epsilon_ ) ) {
187  ++timestep;
188  // Compute all possible outcomes, from our previous results.
189  // This means that for each action-observation pair, we are going
190  // to obtain the same number of possible outcomes as the number
191  // of entries in our initial vector w.
192  const auto projs = projecter(v[timestep-1]);
193  // Here we find the minimum number of VEntries that we need to improve
194  // v on all beliefs from v[timestep-1].
195  v.emplace_back( crossSum( projs, beliefs, v[timestep-1] ) );
196 
197  // Check convergence
198  if ( useEpsilon ) {
199  variation = weakBoundDistance(v[timestep-1], v[timestep]);
200  }
201  }
202 
203  return std::make_tuple(true, v);
204  }
