26 ai::Matrix2D makeA0MatrixStatus(unsigned neighbors, size_t id, double pFailBase, double pFailBonus, double pDeadBase, double pDeadBonus) {
27  const unsigned neighborsCombinations = std::pow(3, neighbors);
28  ai::Matrix2D retval(3 * neighborsCombinations, 3);
29 
30  // We need the PartialFactorsEnumerator since the neighbors ids might be
31  // lower and/or higher than this agent; so in order to iterate correctly to
32  // fill the matrix we rely on the enumerator.
33  aif::PartialFactorsEnumerator e(aif::Factors(neighbors + 1, 3));
34  for (size_t i = 0; e.isValid(); e.advance(), ++i) {
35  double bonus = 0.0;
36  for (size_t n = 0; n < neighbors + 1; ++n) {
37  if (n == id) continue;
38  if (e->second[n] == Fail) bonus += pFailBonus;
39  else if (e->second[n] == Dead) bonus += pDeadBonus;
40  }
41  bonus /= neighbors;
42 
43  const double pFail = pFailBase + bonus;
44  const double pDead = pDeadBase + bonus;
45 
46  // Good Fail Dead
47  if (e->second[id] == Good)
48  retval.row(i) << (1.0 - pFail), pFail, 0.0;
49  else if (e->second[id] == Fail)
50  retval.row(i) << 0.0, (1.0 - pDead), pDead;
51  else if (e->second[id] == Dead)
52  retval.row(i) << 0.0, 0.0, 1.0;
53  }
54 
55  return retval;
56 }

102 afm::CooperativeModel makeSysAdminBiRing(unsigned agents,
103  // Status transition params.
104  double pFailBase, double pFailBonus, double pDeadBase, double pDeadBonus,
105  // Load transition params.
106  double pLoad, double pDoneG, double pDoneF)
107 {
108  // Parameters for this network type:
109  // In a ring we have 2 neighbors.
110  constexpr unsigned neighbors = 2;
111 
112  // We factor the state space into two variables per each agent: status and
113  // load. Each of them can assume 3 different values.
114  aif::State S(agents * 2);
115  std::fill(std::begin(S), std::end(S), 3);
116 
117  // Each agent has a single action, so the size of the action space is equal
118  // to the number of agents.
119  aif::Action A(agents);
120  std::fill(std::begin(A), std::end(A), 2);
121 
122  // All matrices but the a0 status transitions do not depend on the
123  // neighbors, so we can create them only once and just copy them when we
124  // need them.
125  const auto sa1Matrix = makeA1MatrixStatus();
126  const auto la0Matrix = makeA0MatrixLoad(pLoad, pDoneG, pDoneF);
127  const auto la1Matrix = makeA1MatrixStatus();
128 
129  auto ddn = aif::FactoredDDN();
130  for (size_t a = 0; a < agents; ++a) {
131  // Here, for each action, we have to create two transition nodes: one
132  // for the status of the machine, and another for the load.
133  // Both nodes only depend on the action of its agent.
134 
135  // Status node, only depends on the action of 'a'
136  aif::FactoredDDN::Node nodeStatus{{a}, {}};
137 
138  // Status nodes for action 0 (do nothing) and action 1 (restart) respectively.
139  // Node that the transition node for action 0 depends on the neighbors,
140  // since whether they are failing or not affects whether this machine
141  // will fail or not. If we reset, we don't really care.
142  aif::DBN::Node sa0{{}, {}};
143  unsigned neighborId;
144  // Set the correct dependencies for the ring
145  if (a == 0) {
146  sa0.tag = {0, 2, (agents - 1) * 2};
147  neighborId = 0;
148  }
149  else if (a == agents - 1) {
150  sa0.tag = {0, (a - 1) * 2, a * 2};
151  neighborId = 2;
152  }
153  else {
154  sa0.tag = {(a - 1) * 2, a * 2, (a + 1) * 2};
155  neighborId = 1;
156  }
157  sa0.matrix = makeA0MatrixStatus(neighbors, neighborId, pFailBase, pFailBonus, pDeadBase, pDeadBonus);
158 
159  aif::DBN::Node sa1{{a * 2}, sa1Matrix};
160 
161  nodeStatus.nodes.emplace_back(std::move(sa0));
162  nodeStatus.nodes.emplace_back(std::move(sa1));
163 
164  aif::FactoredDDN::Node nodeLoad{{a}, {}};
165 
166  // Here we only depend on our own previous load state
167  aif::DBN::Node la0{{a * 2, (a * 2) + 1}, la0Matrix};
168  aif::DBN::Node la1{{(a * 2) + 1}, la1Matrix};
169 
170  nodeLoad.nodes.emplace_back(std::move(la0));
171  nodeLoad.nodes.emplace_back(std::move(la1));
172 
173  ddn.nodes.emplace_back(std::move(nodeStatus));
174  ddn.nodes.emplace_back(std::move(nodeLoad));
175  }
176 
177  // All reward matrices for all agents are the same, so we build it here
178  // once.
179  //
180  // In particular, we get 1 reward each time we get to a Done state.
181  // However, our matrix of rewards is SxA (with no end states), so we need
182  // to convert our definition of reward into SxA format.
183  //
184  // This means that we need to see which dependencies the Load state has:
185  // both the previous Load and previous Status.
186  ai::Matrix2D rewardMatrix(3 * 3, 2);
187  constexpr double finishReward = 1.0;
188  rewardMatrix.setZero();
189 
190  // Basically, the only way we can get reward is by:
191  // - Starting from the Load state (since it's the only one that can complete)
192  // - Doing action 0;
193  // - And ending up in the Done state.
194  //
195  // Remember that R(s,a) = sum_s1 T(s,a,s1) * R(s,a,s1)
196  rewardMatrix(Load * 3 + Good, 0) = la0Matrix(Load * 3 + Good, Done) * finishReward;
197  rewardMatrix(Load * 3 + Fail, 0) = la0Matrix(Load * 3 + Fail, Done) * finishReward;
198  rewardMatrix(Load * 3 + Dead, 0) = la0Matrix(Load * 3 + Dead, Done) * finishReward;
199 
200  aif::Factored2DMatrix rewards;
201  for (size_t a = 0; a < agents; ++a) {
202  // Now we set all of them with the correct dependencies.
203  aif::BasisMatrix basis;
204  basis.tag = {a * 2, a * 2 + 1}; // We depend on the before status and load;
205  basis.actionTag = {a}; // And on our action.
206  basis.values = rewardMatrix;
207 
208  rewards.bases.emplace_back(std::move(basis));
209  }
210 
211  return afm::CooperativeModel(std::move(S), std::move(A), std::move(ddn), std::move(rewards), 0.95);
212 }
