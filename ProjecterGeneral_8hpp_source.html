95  typename ProjecterGeneral<M>::ProjectionsRow ProjecterGeneral<M>::operator()(const VList & w, const size_t a) {
96  ProjectionsRow projections( boost::extents[O] );
97 
98  for ( size_t o = 0; o < O; ++o ) {
99  // Here we put in just the immediate rewards so that the cross-summing step in the main
100  // function works correctly. However we communicate via the boolean that pruning should
101  // not be done at this step (since adding constants shouldn't do anything anyway).
102  if ( !possibleObservations_[a][o] ) {
103  // We add a parent id anyway in order to keep the code that cross-sums simple. However
104  // note that this fake ID of 0 should never be used, so it should be safe to avoid
105  // setting it to a special value like -1. If one really wants to check, he/she can
106  // just look at the observation table and the belief and see if it makes sense.
107  projections[o].emplace_back(immediateRewards_.row(a), a, VObs(1,0));
108  continue;
109  }
110 
111  // Otherwise we compute a projection for each ValueFunction supplied to us.
112  for ( size_t i = 0; i < w.size(); ++i ) {
113  const auto & v = std::get<VALUES>(w[i]);
114  MDP::Values vproj(S); vproj.fill(0.0);
115  // For each value function in the previous timestep, we compute the new value
116  // if we performed action a and obtained observation o.
117  for ( size_t s = 0; s < S; ++s )
118  // vproj_{a,o}[s] = R(s,a) / |O| + discount * sum_{s'} ( T(s,a,s') * O(s',a,o) * v_{t-1}(s') )
119  for ( size_t s1 = 0; s1 < S; ++s1 )
120  vproj[s] += model_.getTransitionProbability(s,a,s1) * model_.getObservationProbability(s1,a,o) * v[s1];
121  // Set new projection with found value and previous V id.
122  // projections[o].emplace_back(vproj, a, VObs(1,i));
123  projections[o].emplace_back(vproj * discount_ + immediateRewards_.row(a).transpose(), a, VObs(1,i));
124  }
125  }
126 
127  return projections;
128  }
