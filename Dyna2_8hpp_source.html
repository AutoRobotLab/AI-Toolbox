227  void Dyna2<M>::batchUpdateQ(const size_t initS) {
228  transientLearning_.clearTraces();
229 
230  size_t s = initS;
231  size_t a = internalPolicy_->sampleAction(s);
232  for ( unsigned i = 0; i < N; ++i ) {
233  const auto [s1, rew] = model_.sampleSR(s, a);
234  const size_t a1 = internalPolicy_->sampleAction(s1);
235 
236  transientLearning_.stepUpdateQ(s, a, s1, a1, rew);
237 
238  if (model_.isTerminal(s1)) {
239  s = initS;
240  a = internalPolicy_->sampleAction(s);
241  } else {
242  s = s1;
243  a = a1;
244  }
245  }
