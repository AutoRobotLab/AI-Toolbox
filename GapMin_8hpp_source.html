274  // initial lower bound.

278  const auto rend = boost::make_transform_iterator(std::end (lbVList), unwrap);

281  }

284  // The same we do here with FIB for the input POMDP.
285  MDP::QFunction ubQ = std::get<1>(fib(pomdp));
286  AI_LOGGER(AI_SEVERITY_DEBUG, "Initial QFunction:\n" << ubQ);
287 
288  // At the same time, we start initializing fibQ, which will be our
289  // pseudo-alphaVector storage for our belief-POMDPs which we'll create
290  // later.
291  //
292  // The basic idea is to create a new POMDP where each state is a belief
293  // of the input POMDP. This allows us to obtain better upper bounds,
294  // and project them to our input POMDP.
295  auto fibQ = Matrix2D(pomdp.getS()+1, pomdp.getA());
296  fibQ.topLeftCorner(pomdp.getS(), pomdp.getA()).noalias() = ubQ;
297  fibQ.row(pomdp.getS()).noalias() = initialBelief.transpose() * ubQ;
298 
299  // While we store the lower bound as alphaVectors, the upper bound is
300  // composed by both alphaVectors (albeit only S of them - out of the
301  // FastInformedBound), and a series of belief-value pairs, which we'll
302  // use with the later-constructed new POMDP in order to improve our
303  // bounds.
304  UbVType ubV = {
305  {initialBelief}, {fibQ.row(pomdp.getS()).maxCoeff()}
306  };
307 
308  // We also store two numbers for the overall lowerBound/upperBound
309  // differences. They are the values of the lowerBound and the
310  // upperBound at the initial belief.
311  double lb;
312  {
313  const auto rbegin = boost::make_transform_iterator(std::begin(lbVList), unwrap);
314  const auto rend = boost::make_transform_iterator(std::end (lbVList), unwrap);
315  findBestAtPoint(initialBelief, rbegin, rend, &lb);
316  }

318 
319  AI_LOGGER(AI_SEVERITY_INFO, "Initial bounds: " << lb << ", " << ub);
320 
321  while (true) {

324 
325  if (checkEqualSmall(var, 0.0) || var < threshold)
326  break;
327 
328  epsilon_ = threshold * (1.0 - pomdp.getDiscount()) / 2.0;
329  // Now we find beliefs for both lower and upper bound where we
330  // think we can improve. For the ub beliefs we also return their
331  // values, since we need them to improve the ub.
332  auto [newLbBeliefs, newUbBeliefs, newUbVals] = selectReachableBeliefs(pomdp, initialBelief, lbVList, lbBeliefs, ubQ, ubV);
333  const auto newLbBeliefsSize = newLbBeliefs.size();

338  for (const auto & b : newLbBeliefs)

341  // add it to the beliefs we already had, and we rerun PBVI.
342  lbBeliefs.insert(std::end(lbBeliefs), std::make_move_iterator(std::begin(newLbBeliefs)), std::make_move_iterator(std::end(newLbBeliefs)));
343

346  // alphaVectors.
347  auto sol = pbvi(pomdp, lbBeliefs, ValueFunction{std::move(lbVList)});
348 
349  lbVList = std::move(std::get<1>(sol).back());

351  const auto rbegin = boost::make_transform_iterator(std::begin(lbVList), unwrap);
352  const auto rend = boost::make_transform_iterator(std::end (lbVList), unwrap);
353 
354  lbBeliefs.erase(

356  std::begin(lbBeliefs), std::end(lbBeliefs),
357  rbegin, rend
358  ),
359  std::end(lbBeliefs)
360  );
361 
362  // And we recompute the lower bound.
363  findBestAtPoint(initialBelief, rbegin, rend, &lb);
364  }
365  }
366 
367  if (newUbBeliefsSize > 0) {
368  // Here we do the same for the upper bound.
369  const auto prevRows = pomdp.getS() + ubV.first.size();
370  fibQ.conservativeResize(prevRows + newUbBeliefsSize, Eigen::NoChange);
371 
372  AI_LOGGER(AI_SEVERITY_DEBUG, "UB: Adding " << newUbBeliefsSize << " new beliefs...");
373  for (size_t i = 0; i < newUbBeliefsSize; ++i)
374  AI_LOGGER(AI_SEVERITY_DEBUG, "UB: - Belief: " << newUbBeliefs[i].transpose() << " -- value: " << newUbVals[i]);
375 
376  // For each newly found belief which can improve the upper
377  // bound, we add it to to the list containing the beliefs for
378  // the upper bound. At the same time we add horizontal planes
379  // in the fibQ which will come useful on the next round of
380  // FastInformedBound.
381  for (size_t i = 0; i < newUbBeliefs.size(); ++i) {
382  ubV.first.emplace_back(std::move(newUbBeliefs[i]));
383  ubV.second.emplace_back(newUbVals[i]);
384  fibQ.row(prevRows + i).fill(newUbVals[i]);
385  }
386 
387  // We create a new POMDP where each state is a belief.
388  auto [newPOMDP, newPOMDPSOSA] = makeNewPomdp(pomdp, ubQ, ubV);
389  // And we approximate its upper bound.
390  fibQ = std::get<1>(fib(newPOMDP, newPOMDPSOSA, std::move(fibQ)));
391  // We extract from the found upper bound the part for the
392  // states of the input POMDP, and we copy them to our
393  // upperBound alphavectors. We additionally update the values
394  // for all ub beliefs.
395  ubQ.noalias() = fibQ.topRows(pomdp.getS());
396  for (size_t i = 0; i < ubV.second.size(); ++i)
397  ubV.second[i] = fibQ.row(pomdp.getS() + i).maxCoeff();
398 
399  // Finally, we remove some unused stuff, and we recompute the upperbound.
400  cleanUp(ubQ, &ubV, &fibQ);
401 
402  ub = std::get<0>(UB(initialBelief, ubQ, ubV));
403  }
404 
405  // Update the difference between upper and lower bound so we can
406  // return it/use it to stop the loop.
407  auto oldVar = var;
408  var = ub - lb;
409  AI_LOGGER(AI_SEVERITY_INFO, "Updated bounds to " << lb << ", " << ub << " -- size LB: " << lbVList.size() << ", size UB " << ubV.first.size());
410 
411  // Stop if we didn't find anything new, or if we have converged the bounds.
412  if (newLbBeliefsSize + newUbBeliefsSize == 0 || std::fabs(var - oldVar) < epsilon_ * 5)
413  break;
414  }
415  return std::make_tuple(lb, ub, lbVList, ubQ);
416  }

419  std::tuple<GapMin::IntermediatePOMDP, SparseMatrix4D> GapMin::makeNewPomdp(const M& model, const MDP::QFunction & ubQ, const UbVType & ubV) {
420  size_t S = model.getS() + ubV.first.size();
421 
422  // First we build the new reward function. For normal states, this is
423  // the same as the old one. For all additional states (beliefs), we
424  // simply take their expected reward with respect to the original
425  // reward function.
426  Matrix2D R(S, model.getA());
427  const auto & ir = [&]{
428  if constexpr (MDP::is_model_eigen_v<M>) return model.getRewardFunction();
429  else return computeImmediateRewards(model);
430  }();
431 
432  R.topRows(model.getS()) = ir;
433  for (size_t b = 0; b < ubV.first.size(); ++b)
434  R.row(model.getS()+b) = ubV.first[b].transpose() * ir;
435 
436  // Now we create the SOSA matrix for this new POMDP. For each pair of
437  // action/observation, and for each belief we have (thus state), we
438  // compute the probability of going to any other belief.
439  //
440  // This is done through the UB function, although I must admit I don't
441  // fully understand the math behind of why it works.
442  Belief helper(model.getS()), corner(model.getS());
443  corner.fill(0.0);
444 
445  SparseMatrix4D sosa( boost::extents[model.getA()][model.getO()] );
446  const auto updateMatrix = [&](SparseMatrix2D & m, const Belief & b, size_t a, size_t o, size_t index) {
447  updateBeliefUnnormalized(model, b, a, o, &helper);
448  auto sum = helper.sum();
449  if (checkDifferentSmall(sum, 0.0)) {
450  // Note that we do not normalize helper since we'd also have to
451  // multiply `dist` by the same probability. Instead we don't
452  // normalize, and we don't multiply, so we save some work.
453  Vector dist = std::get<1>(UB(helper, ubQ, ubV));
454  for (size_t i = 0; i < S; ++i)
455  if (checkDifferentSmall(dist[i], 0.0))
456  m.insert(index, i) = dist[i];
457  }
458  };
459 
460  for (size_t a = 0; a < model.getA(); ++a) {
461  for (size_t o = 0; o < model.getO(); ++o) {
462  SparseMatrix2D m(S, S);
463  for (size_t s = 0; s < model.getS(); ++s) {
464  corner[s] = 1.0;
465  updateMatrix(m, corner, a, o, s);
466  corner[s] = 0.0;
467  }
468 
469  for (size_t b = 0; b < ubV.first.size(); ++b)
470  updateMatrix(m, ubV.first[b], a, o, model.getS() + b);
471 
472  // After updating all rows of the matrix, we put it inside the
473  // SOSA table.
474  sosa[a][o] = std::move(m);
475  sosa[a][o].makeCompressed();
476  }
477  }
478 
479  // Finally we return a POMDP with no transition nor observation
480  // function, since those are contained in the SOSA matrix.
481  //
482  // We do however include the new reward function that contains rewards
483  // for each new "state"/belief.
484  return std::make_tuple(
485  IntermediatePOMDP(
486  NO_CHECK, model.getO(), Matrix3D(),
487  NO_CHECK, S, model.getA(), Matrix3D(), std::move(R), model.getDiscount()
488  ),
489  std::move(sosa)
490  );
491  }

510  template <typename M, typename = std::enable_if_t<is_model_v<M>>>
511  std::tuple<size_t, double> bestConservativeAction(const M & pomdp, const Belief & initialBelief, const VList & lbVList) {
512  MDP::QFunction ir = [&]{
513  if constexpr (MDP::is_model_eigen_v<M>)
514  return pomdp.getRewardFunction();
515  else
516  return computeImmediateRewards(pomdp);
517  }();
518 
519  Vector bpAlpha(pomdp.getS());
520  for (size_t a = 0; a < pomdp.getA(); ++a) {
521  const Belief intermediateBelief = updateBeliefPartial(pomdp, initialBelief, a);
522 
523  bpAlpha.fill(0.0);
524 
525  for (size_t o = 0; o < pomdp.getO(); ++o) {
526  Belief nextBelief = updateBeliefPartialUnnormalized(pomdp, intermediateBelief, a, o);
527 
528  const auto nextBeliefProbability = nextBelief.sum();
529  if (checkEqualSmall(nextBeliefProbability, 0.0)) continue;
530  // Now normalized
531  nextBelief /= nextBeliefProbability;
532 
533  const auto rbegin = boost::make_transform_iterator(std::begin(lbVList), unwrap);
534  const auto rend = boost::make_transform_iterator(std::end (lbVList), unwrap);
535  auto it = findBestAtPoint(nextBelief, rbegin, rend).base();
536 
537  bpAlpha += pomdp.getObservationFunction(a).col(o).cwiseProduct(it->values);
538  }
539  ir.col(a) += pomdp.getDiscount() * pomdp.getTransitionFunction(a) * bpAlpha;
540  }
541 
542  size_t id;
543  double v = (initialBelief.transpose() * ir).maxCoeff(&id);
544 
545  return std::make_tuple(id, v);
546  }

548  template <typename M, typename>
549  std::tuple<size_t, double> GapMin::bestPromisingAction(const M & pomdp, const Belief & belief, const MDP::QFunction & ubQ, const GapMin::UbVType & ubV) {
550  Vector qvals = belief.transpose() * [&]{
551  if constexpr (MDP::is_model_eigen_v<M>)
552  return pomdp.getRewardFunction();
553  else
554  return computeImmediateRewards(pomdp);
555  }();
556 
557  for (size_t a = 0; a < pomdp.getA(); ++a) {
558  const Belief intermediateBelief = updateBeliefPartial(pomdp, belief, a);
559  double sum = 0.0;
560  for (size_t o = 0; o < pomdp.getO(); ++o) {
561  Belief nextBelief = updateBeliefPartialUnnormalized(pomdp, intermediateBelief, a, o);
562 
563  const auto prob = nextBelief.sum();
564  if (checkEqualSmall(prob, 0.0)) continue;
565  // Note that we do not normalize nextBelief since we'd also
566  // have to multiply the result by the same probability. Instead
567  // we don't normalize, and we don't multiply, so we save some
568  // work.
569  sum += std::get<0>(UB(nextBelief, ubQ, ubV));
570  }
571  qvals[a] += pomdp.getDiscount() * sum;
572  }
573  size_t bestAction;
574  double bestValue = qvals.maxCoeff(&bestAction);
575 
576  return std::make_tuple(bestAction, bestValue);
577  }

579  template <typename M, typename>
580  std::tuple<std::vector<Belief>, std::vector<Belief>, std::vector<double>> GapMin::selectReachableBeliefs(
581  const M & pomdp, const Belief & initialBelief, const VList & lbVList,

590  visitedBeliefs.reserve(maxVisitedBeliefs);

594

597 
598  // We initialize the queue with the initial belief.
599  {

602  const auto rend = boost::make_transform_iterator(std::end (lbVList), unwrap);

605  queue.emplace(QueueElement(initialBelief, 0.0, 1.0, currentLowerBound, currentUpperBound, 1, {}));
606  }
607 
608  while (!queue.empty() && newBeliefs < maxNewBeliefs) {
609  const auto [belief, gap, beliefProbability, currentLowerBound, currentUpperBound, depth, path] = queue.top();
610  (void)gap; // ignore gap variable
611  queue.pop();
612

614  // queue the same belief multiple times. We also limit the size of
615  // the history to avoid the check taking too much time, we tend to
616  // go deeper in the belief tree so it shouldn't be too dangerous.
617  if (visitedBeliefs.size() == maxVisitedBeliefs) {

619  overwriteCounter = (overwriteCounter + 1) % maxVisitedBeliefs;

622  }

625  // upperBound and the lowerBound.

627  // If the found actions improve on the bounds, then we'll add this
628  // belief to the list.

632  // slows the code down - probably cache problems.
633  const auto [ubAction, ubActionValue] = bestPromisingAction(pomdp, belief, ubQ, ubV);
634  const auto [lbAction, lbActionValue] = bestConservativeAction(pomdp, belief, lbVList);
635

638  /***********************
639  ** UPPER GAP **
640  ***********************/

643  // We don't consider corners
644  for (auto i = 0; i < b.size(); ++i)
645  if (checkEqualSmall(b[i], 0.0) || checkEqualSmall(b[i], 1.0))
646  return false;
647

649  const auto check = [&b](const Belief & bb){ return checkEqualProbability(b, bb); };
650  {
651  auto it = std::find_if(std::begin(newUbBeliefs), std::end(newUbBeliefs), check);
652  if (it != std::end(newUbBeliefs))
653  return false;
654  }
655  {
656  auto it = std::find_if(std::begin(ubV.first), std::end(ubV.first), check);
657  if (it != std::end(ubV.first))
658  return false;
659  }
660  return true;
661  };
662 
663  if (validForUb(belief) && ubActionValue < currentUpperBound - epsilon_) {
664  newUbBeliefs.push_back(belief);
665  newUbValues.push_back(ubActionValue);
666

668  // Again, we don't consider corners.
669  for (const auto & p : path) {
670  if (validForUb(p)) {
671  newUbBeliefs.push_back(p);
672  newUbValues.push_back(std::get<0>(UB(p, ubQ, ubV)));
673  }
674  }
675  // Note we only count a single belief even if we added more via
676  // the path (as per original code).
677  ++newBeliefs;
678  }
679 
680  /***********************
681  ** LOWER GAP **
682  ***********************/
683 
684  // For the lower gap we don't care about corners (as per original
685  // code). We still check on the lower bound lists though.
686  const auto validForLb = [&newLbBeliefs, &lbBeliefs](const Belief & b) {
687  const auto check = [&b](const Belief & bb){ return checkEqualProbability(b, bb); };
688  {
689  auto it = std::find_if(std::begin(newLbBeliefs), std::end(newLbBeliefs), check);
690  if (it != std::end(newLbBeliefs))
691  return false;
692  }
693  {
694  auto it = std::find_if(std::begin(lbBeliefs), std::end(lbBeliefs), check);
695  if (it != std::end(lbBeliefs))
696  return false;
697  }
698  return true;
699  };
700 
701  if (validForLb(belief) && lbActionValue > currentLowerBound + epsilon_) {
702  // We add the new belief, and the same is done for all
703  // beliefs that led us to this one (if they're valid -
704  // i.e., we didn't have them already).
705  newLbBeliefs.push_back(belief);
706 
707  for (const auto & p : path) {
708  if (validForLb(p)) {
709  newLbBeliefs.push_back(p);
710  }
711  }
712  // Note we only count a single belief even if we added more via
713  // the path (as per original code).
714  ++newBeliefs;
715  }
716 
717  /***********************
718  ** QUEUE EXPANSION **
719  ***********************/
720 
721  // Avoid it if we're already done anyway.
722  if (newBeliefs >= maxNewBeliefs)
723  break;
724 
725  auto newPath = path;
726  newPath.push_back(belief);
727 
728  // For each new possible belief, we look if we've already visited
729  // it. If not, we compute the gap at that point, and we add it to
730  // the queue.
