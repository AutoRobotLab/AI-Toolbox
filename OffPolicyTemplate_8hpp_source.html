358  for (size_t a = 0; a < A; ++a)

367  template <typename Derived>
368  void OffPolicyControl<Derived>::stepUpdateQ(const size_t s, const size_t a, const size_t s1, const double rew) {
369  // The basic idea here is that differently from the evaluation, we want
370  // to do a maximization. At the same time, to work we need to "round
371  // the edges" a bit, and that's why we assume an epsilon-greedy policy.
372  //
373  // The expected value we can compute easily since each action has the
374  // same probability of being chosen, but for the greedy one which is
375  // more likely.
376  size_t maxA;
377  double expectedQ = 0.0;
378  double maxV = std::numeric_limits<double>::lowest();
379  for (size_t aa = 0; aa < A; ++aa) {
380  expectedQ += q_(s1, aa);
381  if (maxV < q_(s1, aa)) {
382  maxA = aa;
383  maxV = q_(s1, aa);
384  }
385  }
386  expectedQ *= (1.0 - exploration_) / A;
387  expectedQ += maxV * exploration_;
388 
389  const auto error = alpha_ * ( rew + discount_ * expectedQ - q_(s, a) );
390  const auto traceDiscount = discount_ * static_cast<Derived*>(this)->getTraceDiscount(s, a, s1, rew, maxA);
391 
392  updateTraces(s, a, error, traceDiscount);
393  }
