157  template <typename PIterator, typename VIterator>
158  PIterator extractBestUsefulPoints(PIterator pbegin, PIterator pend, VIterator begin, VIterator end) {
159  const auto pointsN = std::distance(pbegin, pend);
160  const auto entriesN = std::distance(begin, end);
161 
162  std::vector<std::pair<PIterator, double>> bestValues(entriesN, {pend, std::numeric_limits<double>::lowest()});
163  const auto maxBound = pointsN < entriesN ? pend : pbegin + entriesN;
164 
165  // So the idea here is that we advance IT only if we found a Point
166  // which supports a previously unsupported Hyperplane. This allows us
167  // to avoid doing later work for compacting the points before the
168  // bound.
169  //
170  // If instead the found Point takes into consideration an already
171  // supported Hyperplane, then it either is better or not. If it's
172  // better, we swap it with whatever was before. In both cases, the
173  // Point to discard ends up at the end and we decrease the bound.
174  auto it = pbegin;
175  auto bound = pend;
176  while (it < bound && it < maxBound) {
177  double value;
178  const auto vId = std::distance(begin, findBestAtPoint(*it, begin, end, &value));
179  if (bestValues[vId].second < value) {
180  if (bestValues[vId].first == pend) {
181  bestValues[vId] = {it++, value};
182  continue;
183  } else {
184  bestValues[vId].second = value;
185  iter_swap(bestValues[vId].first, it);
186  }
187  }
188  iter_swap(it, --bound);
189  }
190  if (it == bound) return it;
191 
192  // If all Hyperplanes have been supported by at least one Point, then
193  // we can finish up the rest with less swaps and checks. Here we only
194  // swap with the best if needed, otherwise we don't have to do
195  // anything.
196  //
197  // This is because we can return one Point per Hyperplane at the most,
198  // so if we're here the bound is not going to move anyway.
199  while (it < bound) {
200  double value;
201  const auto vId = std::distance(begin, findBestAtPoint(*it, begin, end, &value));
202  if (bestValues[vId].second < value) {
203  bestValues[vId].second = value;
204  iter_swap(bestValues[vId].first, it);
205  }
206  ++it;
207  }
208  return maxBound;
209  }

242  std::vector<std::pair<Point, double>> findVerticesNaive(NewIt beginNew, NewIt endNew, OldIt alphasBegin, OldIt alphasEnd) {
243  std::vector<std::pair<Point, double>> vertices;
244 
245  const size_t alphasSize = std::distance(alphasBegin, alphasEnd);
246  if (alphasSize == 0) return vertices;
247  const size_t S = alphasBegin->size();
248 
249  // This enumerator allows us to compute all possible subsets of S-1
250  // elements. We use it on both the alphas, and the boundaries, thus the
251  // number of elements we iterate over is alphasSize + S.
252  SubsetEnumerator enumerator(S - 1, 0ul, alphasSize + S);
253 
254  // This is the matrix on the left side of Ax = b (where A is m)
255  Matrix2D m(S + 1, S + 1);
256  m.row(0)[S] = -1; // First row is always a vector
257 
258  Vector boundary(S+1);
259  boundary[S] = 0.0; // The boundary doesn't care about the value
260 
261  // This is the vector on the right side of Ax = b
262  Vector b(S+1); b.setZero();
263 
264  Vector result(S+1);
265 
266  // Common matrix/vector setups
267 
268  for (auto newVIt = beginNew; newVIt != endNew; ++newVIt) {
269  m.row(0).head(S) = *newVIt;
270 
271  enumerator.reset();
272 
273  // Get subset of planes, find corner with LU
274  size_t last = 0;
275  while (enumerator.isValid()) {
276  // Reset boundaries to care about all dimensions
277  boundary.head(S).fill(1.0);
278  size_t counter = 1;
279  // Note that we start from last to avoid re-copying vectors
280  // that are already in the matrix in their correct place.
281  for (auto i = last; i < enumerator->size(); ++i) {
282  // For each value in the enumerator, if it is less than
283  // alphasSize it is referring to an alphaVector we need to
284  // take into account.
285  const auto index = (*enumerator)[i];
286  if (index < alphasSize) {
287  // Copy the right vector in the matrix.
288  m.row(counter).head(S) = *std::next(alphasBegin, index);
289  m.row(counter)[S] = -1;
290  ++counter;
291  } else {
292  // We limit the index-th dimension (minus alphasSize to scale in a 0-S range)
293  boundary[index - alphasSize] = 0.0;
294  }
295  }
296  m.row(counter) = boundary;
297  b[counter] = 1.0;
298  ++counter;
299 
300  // Note that we only need to consider the first "counter" rows,
301  // as the boundaries get merged in a single one.
302  result = m.topRows(counter).colPivHouseholderQr().solve(b.head(counter));
303 
304  b[counter-1] = 0.0;
305 
306  // Add to found only if valid, otherwise skip.
307  if (((result.head(S).array() >= 0) && (result.head(S).array() <= 1.0)).all())
308  vertices.emplace_back(result.head(S), result[S]);
309 
310  // Advance, and take the id of the first index changed in the
311  // next combination.
312  last = enumerator.advance();
313 
314  // If the index went over the alpha list, then we'd only have
315  // boundaries, but we don't care about those cases (since we
316  // assume we already have the corners of the simplex computed).
317  // Thus, terminate.
318  if ((*enumerator)[last] >= alphasSize) break;
319  }
320  }
321  return vertices;
322  }

343  template <typename It>
344  double computeOptimisticValue(const Point & p, It pvBegin, It pvEnd) {
345  const size_t vertexNumber = std::distance(pvBegin, pvEnd);
346  if (vertexNumber == 0) return 0.0;
347  const size_t S = p.size();
348 
349  LP lp(S);
350 
351  /*
352  * With this LP we are looking for an optimistic hyperplane that can
353  * tightly fit all corners that we already have, and maximize the value
354  * at the input point.
355  *
356  * Our constraints are of the form
357  *
358  * vertex[0][0]) * h0 + vertex[0][1]) * h1 + ... <= vertex[0].currentValue
359  * vertex[1][0]) * h0 + vertex[1][1]) * h1 + ... <= vertex[1].currentValue
360  * ...
361  *
362  * Since we are looking for an optimistic hyperplane, all variables are
363  * unbounded since the hyperplane may need to go negative at some
364  * states.
365  *
366  * Finally, our constraint is a row to maximize:
367  *
368  * b * v0 + b * v1 + ...
369  *
370  * Which means we try to maximize the value of the input point with the
371  * newly found hyperplane.
372  */
373 
374  // Set objective to maximize
375  lp.row = p;
376  lp.setObjective(true);
377 
378  // Set unconstrained to all variables
379  for (size_t s = 0; s < S; ++s)
380  lp.setUnbounded(s);
381 
382  // Set constraints for all input points and current values.
383  for (auto it = pvBegin; it != pvEnd; ++it) {
384  lp.row = it->first;
385  lp.pushRow(LP::Constraint::LessEqual, it->second);
386  }
387 
388  double retval;
389  // Note that we don't care about the optimistic alphavector, so we
390  // discard it. We check that everything went fine though, in theory
391  // there shouldn't be any problems here.
392  auto solution = lp.solve(0, &retval);
393  assert(solution);
394 
395  return retval;
396  }
