133  ProbabilityVector makeRandomProbability(const size_t S, G & generator) {
134  static std::uniform_real_distribution<double> sampleDistribution(0.0, 1.0);
135  ProbabilityVector b(S);
136  double * bData = b.data();
137  // The way this works is that we're going to generate S-1 numbers in
138  // [0,1], and sort them with together with an implied 0.0 and 1.0, for
139  // a total of S+1 numbers.
140  //
141  // The output will be represented by the differences between each pair
142  // of numbers, after sorting the original vector.
143  //
144  // The idea is basically to take a unit vector and cut it up into
145  // random parts. The size of each part is the value of an entry of the
146  // output.
147 
148  // We must set the first element to zero even if we're later
149  // overwriting it. This is to avoid bugs in case the input S is one -
150  // in which case we should return a vector with a single element
151  // containing 1.0.
152  bData[0] = 0.0;
153  for ( size_t s = 0; s < S-1; ++s )
154  bData[s] = sampleDistribution(generator);
155 
156  // Sort all but the implied last 1.0 which we'll add later.
157  std::sort(bData, bData + S - 1);
158 
159  // For each number, keep track of what was in the vector there, and
160  // transform it into the difference with its predecessor.
161  double helper1 = bData[0], helper2;
162  for ( size_t s = 1; s < S - 1; ++s ) {
163  helper2 = bData[s];
164  bData[s] -= helper1;
165  helper1 = helper2;
166  }
167  // The final one is computed with respect to the overall sum of 1.0.
168  bData[S-1] = 1.0 - helper1;
169 
170  return b;
171  }

217  retval[i] = 1.0;

219  sum += v[i];
220  }
221  }

223  if (checkEqualSmall(sum, 0.0)) {
224  // Any solution here would do, but this seems nice.
225  retval.array() += 1.0 / v.size();
226  } else if (sum > 1.0) {
227  // We normalize the vector.
228  retval.array() *= v.array() / sum;
229  } else {

232  retval.array() *= (v.array() + diff);
