167  ProbabilityVector makeRandomProbability(const size_t S, G & generator) {
168  static std::uniform_real_distribution<double> sampleDistribution(0.0, 1.0);
169  ProbabilityVector b(S);
170  double * bData = b.data();
171  // The way this works is that we're going to generate S-1 numbers in
172  // [0,1], and sort them with together with an implied 0.0 and 1.0, for
173  // a total of S+1 numbers.
174  //
175  // The output will be represented by the differences between each pair
176  // of numbers, after sorting the original vector.
177  //
178  // The idea is basically to take a unit vector and cut it up into
179  // random parts. The size of each part is the value of an entry of the
180  // output.
181 
182  // We must set the first element to zero even if we're later
183  // overwriting it. This is to avoid bugs in case the input S is one -
184  // in which case we should return a vector with a single element
185  // containing 1.0.
186  bData[0] = 0.0;
187  for ( size_t s = 0; s < S-1; ++s )
188  bData[s] = sampleDistribution(generator);
189 
190  // Sort all but the implied last 1.0 which we'll add later.
191  std::sort(bData, bData + S - 1);
192 
193  // For each number, keep track of what was in the vector there, and
194  // transform it into the difference with its predecessor.
195  double helper1 = bData[0], helper2;
196  for ( size_t s = 1; s < S - 1; ++s ) {
197  helper2 = bData[s];
198  bData[s] -= helper1;
199  helper1 = helper2;
200  }
201  // The final one is computed with respect to the overall sum of 1.0.
202  bData[S-1] = 1.0 - helper1;
203 
204  return b;
205  }
