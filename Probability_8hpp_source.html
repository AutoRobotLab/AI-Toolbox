132  ProbabilityVector makeRandomProbability(const size_t S, G & generator) {
133  ProbabilityVector b(S);
134  double * bData = b.data();
135  // The way this works is that we're going to generate S-1 numbers in
136  // [0,1], and sort them with together with an implied 0.0 and 1.0, for
137  // a total of S+1 numbers.
138  //
139  // The output will be represented by the differences between each pair
140  // of numbers, after sorting the original vector.
141  //
142  // The idea is basically to take a unit vector and cut it up into
143  // random parts. The size of each part is the value of an entry of the
144  // output.
145 
146  // We must set the first element to zero even if we're later
147  // overwriting it. This is to avoid bugs in case the input S is one -
148  // in which case we should return a vector with a single element
149  // containing 1.0.
150  bData[0] = 0.0;
151  for ( size_t s = 0; s < S-1; ++s )
152  bData[s] = probabilityDistribution(generator);
153 
154  // Sort all but the implied last 1.0 which we'll add later.
155  std::sort(bData, bData + S - 1);
156 
157  // For each number, keep track of what was in the vector there, and
158  // transform it into the difference with its predecessor.
159  double helper1 = bData[0], helper2;
160  for ( size_t s = 1; s < S - 1; ++s ) {
161  helper2 = bData[s];
162  bData[s] -= helper1;
163  helper1 = helper2;
164  }
165  // The final one is computed with respect to the overall sum of 1.0.
166  bData[S-1] = 1.0 - helper1;
167 
168  return b;
169  }
