587  template <typename BIterator, typename VIterator>
588  BIterator extractBestUsefulBeliefs(BIterator bbegin, BIterator bend, VIterator begin, VIterator end) {
589  const auto beliefsN = std::distance(bbegin, bend);
590  const auto entriesN = std::distance(begin, end);
591 
592  std::vector<std::pair<BIterator, double>> bestValues(entriesN, {bend, std::numeric_limits<double>::lowest()});
593  const auto maxBound = beliefsN < entriesN ? bend : bbegin + entriesN;
594 
595  // So the idea here is that we advance IT only if we found a belief
596  // which supports a previously unsupported VEntry. This allows us to
597  // avoid doing later work for compacting the beliefs before the bound.
598  //
599  // If instead the found belief takes into consideration an already
600  // supported VEntry, then it either is better or not. If it's better,
601  // we swap it with whatever was before. In both cases, the belief to
602  // discard ends up at the end and we decrease the bound.
603  auto it = bbegin;
604  auto bound = bend;
605  while (it < bound && it < maxBound) {
606  double value;
607  const auto vId = std::distance(begin, findBestAtBelief(*it, begin, end, &value));
608  if (bestValues[vId].second < value) {
609  if (bestValues[vId].first == bend) {
610  bestValues[vId] = {it++, value};
611  continue;
612  } else {
613  bestValues[vId].second = value;
614  std::iter_swap(bestValues[vId].first, it);
615  }
616  }
617  std::iter_swap(it, --bound);
618  }
619  if (it == bound) return it;
620 
621  // If all VEntries have been supported by at least one belief, then we
622  // can finish up the rest with less swaps and checks. Here we only swap
623  // with the best if needed, otherwise we don't have to do anything.
624  //
625  // This is because we can return one belief per VEntry at the most, so
626  // if we're here the bound is not going to move anyway.
627  while (it < bound) {
628  double value;
629  const auto vId = std::distance(begin, findBestAtBelief(*it, begin, end, &value));
630  if (bestValues[vId].second < value) {
631  bestValues[vId].second = value;
632  std::iter_swap(bestValues[vId].first, it);
633  }
634  ++it;
635  }
636  return maxBound;
637  }
638 }
