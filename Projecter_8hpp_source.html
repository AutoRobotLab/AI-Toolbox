94  typename Projecter<M>::ProjectionsRow Projecter<M>::operator()(const VList & w, size_t a) {
95  ProjectionsRow projections( boost::extents[O] );
96 
97  for ( size_t o = 0; o < O; ++o ) {
98  // Here we put in just the immediate rewards so that the cross-summing step in the main
99  // function works correctly. However we communicate via the boolean that pruning should
100  // not be done at this step (since adding constants shouldn't do anything anyway).
101  if ( !possibleObservations_[a][o] ) {
102  MDP::Values vproj(S, 0.0);
103  for ( size_t s = 0; s < S; ++s )
104  vproj[s] += immediateRewards_[a][s];
105  // We add a parent id anyway in order to keep the code that cross-sums simple. However
106  // note that this fake ID of 0 should never be used, so it should be safe to avoid
107  // setting it to a special value like -1. If one really wants to check, he/she can
108  // just look at the observation table and the belief and see if it makes sense.
109  projections[o].emplace_back(std::move(vproj), a, VObs(1,0));
110  continue;
111  }
112 
113  // Otherwise we compute a projection for each ValueFunction supplied to us.
114  for ( size_t i = 0; i < w.size(); ++i ) {
115  auto & v = std::get<VALUES>(w[i]);
116  MDP::Values vproj(S, 0.0);
117  // For each value function in the previous timestep, we compute the new value
118  // if we performed action a and obtained observation o.
119  for ( size_t s = 0; s < S; ++s ) {
120  // vproj_{a,o}[s] = R(s,a) / |O| + discount * sum_{s'} ( T(s,a,s') * O(s',a,o) * v_{t-1}(s') )
121  for ( size_t s1 = 0; s1 < S; ++s1 )
122  vproj[s] += model_.getTransitionProbability(s,a,s1) * model_.getObservationProbability(s1,a,o) * v[s1];
123 
124  vproj[s] *= discount_;
125  vproj[s] += immediateRewards_[a][s];
126  }
127  // Set new projection with found value and previous V id.
128  projections[o].emplace_back(std::move(vproj), a, VObs(1,i));
129  }
130  }
131 
132  return projections;
133  }
