121  // bound for each action is computed assuming to take the same action forever
122  // (so the bound for action 0 assumes to forever take action 0, the bound for
123  // action 1 assumes to take action 1, etc.).

131  auto oldAlpha = Vector(m.getS());
132  // Note that here we can take the minimum for each action
133  // separately, since the implied policy will take that action
134  // forever anyway so there cannot be "cross-pollination" between
135  // different actions.

138  else

143  while ( timestep < horizon_ && ( !useEpsilon || variation > epsilon_ ) ) {
144  ++timestep;

146  newAlpha = ir.row(a) + (m.getDiscount() * m.getTransitionFunction(a) * oldAlpha).transpose();
