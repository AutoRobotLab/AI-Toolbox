7 // The gist of this problem is a small grid where
8 // the upper-left corner and the bottom-right corner
9 // are self-absorbing states. The agent can move in a
10 // top-left-down-right way, where each transition that
11 // is not self absorbing results in a reward penalty of -1.
12 // In addition the movements are not guaranteed: the
13 // agent succeeds only 80% of the time.
14 //
15 // Thus the agent needs to be able to find the shortest

17 // state.

46 
47  size_t S = grid.getSizeX() * grid.getSizeY(), A = 4;
48 
49  AIToolbox::Table3D transitions(boost::extents[S][A][S]);
50  AIToolbox::Table3D rewards(boost::extents[S][A][S]);
51 
52  for ( int x = 0; x < 4; ++x ) {
53  for ( int y = 0; y < 4; ++y ) {
54  auto s = grid(x,y);
55  if ( s == 0 || s == S-1 ) {
56  // Self absorbing states
57  for ( size_t a = 0; a < A; ++a )
58  transitions[s][a][s] = 1.0;
59  }
60  else {
61  for ( size_t a = 0; a < A; ++a ) {
62  auto s1 = grid(s);
63  s1.setAdjacent((Direction)a);
64  // If the move takes you outside the map, it doesn't do
65  // anything
66  if ( s == s1 ) transitions[s][a][s1] = 1.0;
67  else {
68  transitions[s][a][s1] = stepUncertainty;
69  transitions[s][a][s] = 1.0 - stepUncertainty;
70  }
71  rewards[s][a][s1] = -1.0;
72  }
73  }
74  }
75  }
